\section{Background}
Many real-world routing and scheduling problems are subject to uncertainty. In a delivery service, for example, a courier must plan a route through several customer locations, yet some customers may have no packages to receive, or their requests may be canceled.
A classical abstraction is the stochastic
traveling salesperson problem (stochastic TSP), where each vertex $v$ comes with a specified probability $p_v$ and becomes \textit{active} (requires service) independently with probability $p_v$. The goal is to find a tour that visits all active vertices and minimizes the expected travel cost, where the cost comes from the underlying metric on vertices.

An important modeling choice is whether the salesperson is allowed to adapt their future decisions based on information revealed during the execution of the tour. In the nonadaptive (a priori) setting, the salesperson commits to a fixed tour before observing the realization; during execution the salesperson follows this tour while shortcutting inactive vertices. In the adaptive setting, decisions may depend on observations revealed along the way, allowing the tour to be adjusted on the fly.

A fundamental question is how much adaptivity can improve performance relative to an optimal nonadaptive tour.
A related algorithmic question is whether we can design provably good adaptive policies using limited computational resources.
Note that computing an optimal adaptive policy (or an optimal a priori tour) for stochastic TSP is computationally hard, since the setting with $p_v = 1$ for all $v \in V$ corresponds to the classical TSP which is known to be APX-hard\cite{Karpinski2015}.
Also, describing an optimal adaptive policy might require exponential space.
Thus, we will be interested in designing adaptive policies that can be implemented efficiently
and whose expected cost can be provably bounded in terms of the optimal cost within some small factor $\alpha \geq 1$.

\section{Notations and Definitions}
Let $(\{r\}\cup V,d)$ be a finite metric space on $n+1$ vertices (a depot vertex $r$ and $n$ clients) equipped with a distance function $d$.
That is, $d$ satisfies the triangle inequality: for any three vertices $u,v,w$, $d(u,v) \leq d(u,w) + d(w,v)$.
A metric is said to be \emph{symmetric} if $d(x, y) = d(y, x)$ for all $x, y$.

A \emph{tour} on a subset $S \subseteq \{r\}\cup V$ containing the depot is a cyclic sequence $T=(r,u_1,u_2,\ldots,u_{|S|-1},r)$ that visits every vertex in $S$ exactly once. Its \emph{cost} is
\[
\cost(T) := d(r,u_1) + d(u_1,u_2) + \cdots + d(u_{|S|-2},u_{|S|-1}) + d(u_{|S|-1},r).
\]
For a subset $A\subseteq V$, we write $\opt(A)$ for the minimum cost of a tour on $\{r\}\cup A$.

In the stochastic setting, the set of active vertices is a subset $A \subseteq V$ generated by independent activations: each vertex $v \in V$ is active (i.e., $v \in A$) independently with probability $p_v$. We denote by $q$ the resulting product distribution on $2^V$, so $q(A) = \prod_{v \in A}p_v \cdot \prod_{v \notin A}(1 - p_v)$.

We consider three benchmarks for stochastic TSP, distinguished by how much the solution may depend on the realization of the active set. In each case we define the cost of a feasible solution via an overloaded function $\cost(\cdot)$.
\begin{enumerate}
    \item \textbf{A posteriori TSP.}
        An \emph{a posteriori solution} is a family $\mathcal{S}=(T_A)_{A\subseteq V}$, where $T_A$ is a tour on $\{r\}\cup A$ for each activation set $A$.
        We define
        \[
        \cost(\mathcal{S}) := \E_{A\sim q}\bigl[\cost(T_A)\bigr].
        \]
        Since each $T_A$ can be chosen independently, the optimum is
        \[
        \optpost := \min_{\mathcal{S}}\,\cost(\mathcal{S}) = \E_{A\sim q}\bigl[\opt(A)\bigr].
        \]

    \item \textbf{A priori (nonadaptive) TSP.}
        An \emph{a priori tour} is a tour $T=(r,v_1,\ldots,v_n,r)$ on the full vertex set $\{r\}\cup V$.
        For a realization $A$, the \emph{shortcut} $T\vert_A$ is obtained by retaining only the vertices in $\{r\}\cup A$ in the cyclic order prescribed by $T$.
        More precisely, if $1 \leq i_1 < i_2 < \cdots < i_{|A|} \leq n$ are such that $\{v_{i_k} : 1 \leq k \leq |A|\} = A$, then $T\vert_A = (r,v_{i_1},v_{i_2},\ldots,v_{i_{|A|}},r)$.
        We overload:
        \[
        \cost(T) := \E_{A\sim q}\bigl[\cost(T\vert_A)\bigr].
        \]
        The optimum is $\optprior := \min_{T}\,\cost(T)$, where the minimum is over all a priori tours.

    \item \textbf{Adaptive TSP.}
        The salesperson knows all the activation probabilities but not the realization $A$.
        During execution, the salesperson calls clients one by one:
        if the called client $v$ is in $A$, the salesperson \emph{must} travel to $v$ immediately; otherwise the salesperson remains in place.
        The order of calls may depend on the outcomes observed so far.

        Formally, an adaptive policy is a function $\pi: (\{r\} \cup V) \times 2^V \rightarrow \{r\} \cup V \cup \{\bot\}$, where $\pi(u,S)$ denotes the next vertex to call when the salesperson is at $u$ and the set of uncalled vertices is $S$.
        We require: (a)~$\pi(u,S) := \bot$ whenever $u \in S$ (invalid input); (b)~$\pi(u,S) = r$ iff $S = \emptyset$ (return to depot only when done); (c)~$\pi(u,S) \in S$ iff $S \neq \emptyset$ (must call an uncalled vertex).

        Given a realization $A$, the policy induces a random tour $T(A;\pi)$ on $\{r\}\cup A$ as follows.
        Initialize $u \leftarrow r$, $S \leftarrow V$.
        While $S \neq \emptyset$: compute $v = \pi(u,S)$; if $v \in A$ then travel to $v$ and set $u\leftarrow v$; set $S \leftarrow S \setminus \{v\}$.
        Finally, return to $r$.
        We overload:
        \[
        \cost(\pi) := \E_{A\sim q}\bigl[\cost(T(A;\pi))\bigr].
        \]
        The optimum is $\optadapt := \min_{\pi}\,\cost(\pi)$, where the minimum is over all adaptive policies.

\end{enumerate}

\section{Research Question}
Among the three benchmarks, a posteriori TSP has the most information (the full realization $A$), a priori TSP has the least (only the metric and probabilities), and adaptive TSP lies in between.

We define an \emph{instance} $\mathcal{I} := (V \cup \{r\}, d, \{p_v\})$ of stochastic TSP. For any instance $\mathcal{I}$,
\begin{equation}\label{eq:optineq}
\optpost \le \optadapt \le \optprior
\end{equation}
since from left to right the solution has less information about the realization, resulting in higher optimal cost.

We define the following gap measures:
\begin{itemize}
    \item \emph{Adaptivity gap}: $\sup_{\mathcal{I}} \f{\optprior}{\optadapt}$ — the worst-case benefit of adaptivity over a fixed a priori tour.
    \item \emph{Clairvoyance gap}: $\sup_{\mathcal{I}} \f{\optadapt}{\optpost}$ — the benefit of full clairvoyance (knowing $A$ upfront) over the best adaptive policy.
    \item \emph{Obliviousness gap}: $\sup_{\mathcal{I}} \f{\optprior}{\optpost}$ — combines both effects.
\end{itemize}
We remark that the literature on a priori TSP uses the term adaptivity gap for what we call the obliviousness gap; we deviate from this convention following \cite{Christalla2025} for the sake of clarity.

For symmetric metrics, the obliviousness gap is known to be at most $O(1)$; in particular, $\optprior/\optpost \le 3$ is shown in \cite{Shmoys2008}.
The same upper bound carries over to the clairvoyance gap and the adaptivity gap by~\eqref{eq:optineq}.
One of the objectives of this project is to obtain better upper and lower bounds on the clairvoyance gap and the adaptivity gap.

For asymmetric metrics, Christalla, Puhlmann and Traub \cite{Christalla2025} show that the obliviousness gap is $O(\sqrt{n})$ and $\Omega(n^{1/4} \cdot \log^{-1}n)$.
Moreover, the construction in Section~3.1 of \cite{Christalla2025} also implies an adaptivity gap of $\Omega(n^{1/4} \cdot \log^{-1}n)$.
Another key objective of this project is to determine whether the clairvoyance gap for asymmetric metrics is $O(1)$. In case of a positive resolution, a further goal is to design an adaptive policy matching this gap up to constant factors. Otherwise, the aim is to construct an infinite family of instances for which the clairvoyance gap grows without bound.

\section{Proof of epsilon normalization equivalence}
We call an instance $I = (V, c, p)$ $\varepsilon$-normalized if there exists a depot $d \in V$ with $p(d) = 1$ and $p(v) = \varepsilon$ for all $v \in V \setminus \{d\}$.

\begin{lemma}\label{lem:transform}
    There is a transform $g$ mapping any instance $I$ to an $\varepsilon$-normalized instance $I' = g(I)$ such that the following hold:
    \begin{alignat*}{2}
        (1-\xi)\optpost(I)  &\le{}& \optpost(I')  &\le{} \optpost(I) \\
        (1-\xi)\optprior(I) &\le{}& \optprior(I') &\le{} \optprior(I) \\
        (1-\xi)\optadapt(I) &\le{}& \optadapt(I') &\le{} \optadapt(I)
    \end{alignat*}
\end{lemma}

There is a lemma from paper \cite{Blauth2024} Lemma 32 which says:

\begin{lemma}[Existence of proper parameters]\label{lem:param}
    For  $\sigma \in (0, 1), \lambda > 0$ and $p \in (0, 1]$. There is $\varepsilon_p \in (0, 1)$ such that for any $\varepsilon \in (0, \varepsilon_p ]$ there is $k \in \mathbb{N}$ such that 
    \begin{enumerate}
        \item $1 - (1-\varepsilon)^k \le p$
        \item $(1-\lambda)p \le 1-(1-\varepsilon)^k$
    \end{enumerate}
\end{lemma}

\begin{proof}[Proof of Lemma \ref{lem:param}]
    We can simplify the inequalities into $1 - p \le (1-\varepsilon)^k \le 1 - (1-\lambda)p$. Suppose $p < 1$, then for any certain $\varepsilon$, the inequalities are equivalent to $\frac{\ln (1 - (1-\lambda)p)}{\ln (1 - \varepsilon)} \le k \le \frac{\ln (1 - p)}{\ln (1 - \varepsilon)}$. Since $k$ has to be an integer, we can choose $\varepsilon_p$ to be small enough such that $\frac{\ln (1 - p)}{\ln (1 - \varepsilon_p)} - \frac{\ln (1 - (1-\lambda)p)}{\ln (1 - \varepsilon_p)} > 1$. Suppose $p = 1$, then we only need to choose $k$ to be large enough such that $(1-\epsilon)^k \le \lambda$. Thus in this case $\varepsilon_p$ can be any value in $(0,1)$.
\end{proof}

\section{Estimate of Rounds Needed}\label{sec:gridexample}

We consider a new graph where vertices are also placed in a $k$ by $k$ grid where $k = \Delta^\Delta$ for some parameter $\Delta$. Instead of connecting all vertices from column $i$ to column $i + 1$, the $j$-th vertex in $i$-th column will only connect to vertices $j'$ in the $i+1$-th column only when under $\Delta$ base, $j'$ differs from  $j$  in at most $1$ coordinate. Here $\Delta$ is a parameter that we firstly set to be $\frac{\log k}{\log \log k}$.

We say two vertices are connected if the distance between them is the same as the (cyclic) distance of their underlying column. It is then easy to see that any active vertex in $i$-th column are connected to any vertex in $i'$-th column if $i' \ge i + \Delta$. We call a path to have $1$ round if it starts from $i$-th column, and after moving only by connecting edges (the edges connecting  two \emph{connected} vertices) it comes across $i$-th column again. Clearly each round costs $\Theta(k)$, we want to know how many rounds do we need to cover all active vertices, which gives an upper bound of optimal a posteriori cost.

Denote $M := \max_{i} \sum_{r=-\Delta}^{\Delta} X_{i+r}$, where $X_i$ represents the number of active vertices in $i$-th column. Then we claim that we need at most $M$ rounds to cover all vertices.

We prove the claim by contradiction. Suppose after $M$ rounds, there are still a vertex $v$ in column $j$ which is not covered. Then consider the interval $I_j$ consisting of columns $[j - \Delta, j + \Delta]$, there are no more than $M$ active vertices in $I_j$, so within the $M$ rounds, there exists at least one round $r$ such that no vertices from $I_j$ is touched (by pigeonhole principle). Then we can adjust the previous path by adding vertex $j$ into round $r$. This is always possible because in the round $r$ we can always find the last vertex before $I_j$ and the first vertex after $I_j$, we can insert vertex $v$ without breaking the connection rule since   any active vertex in $i$-th column are connected to any vertex in $i'$-th column if $i' \ge i + \Delta$. We can always perform this adjustment until there is no such vertex exists, then we have a contradiction.

We then need to bound $\E[M]$ and the optimal a posteriori cost would be bounded by $\E[M] \cdot k$.

Define the cyclic window length $m := 2\Delta + 1$.
Assume $X_1, \dots, X_k$ are independent and
$X_i \sim \textbf{Bin}(k, 1/k)$.
Define the cyclic window sums
$S_i := \sum_{r=0}^{m-1} X_{i+r}$
(indices mod~$k$).
The quantity of interest is $M := \max_{1 \le i \le k} S_i$.

By the trick discussed, we know for any $T$:

\begin{align*}
    \E[M] \le T + \sum_{i=1}^k \E[(S_i - T)^+]
\end{align*}

This is nice because we do not need independence here (we do not have independence since intervals $S_i$ might be overlapping) and by symmetry, we have:

\begin{align*}
    \E[M] \le T + k \E[(S_i - T)^+]
\end{align*}

Then we convert the $\E[(S_1 - T)^+]$ into a tail sum. Since $S_i$ is integer-valued variable, so $\E[(S_1 - T)^+] = \sum_{u=1}^{\infty} \Pr[S_1 \ge T + u]$. So we have:

\begin{align*}
    \E[M] \le T + k \sum_{u=1}^{\infty} \Pr[S_1 \ge T + u]
\end{align*}

We know $S_1 \sim \textbf{Bin}(mk, 1/k)$ with mean $m$ and for large $k$ it is approximated by $\textbf{Poi}(m)$. So let $Z \sim \textbf{Poi}(m)$ and then $\Pr[S_1 \ge t] \simeq \Pr[Z \ge t]$. And for Poisson distribution, we have the property that $\Pr[Z \ge t + 1] \le \frac{m}{t + 1}\Pr[Z \ge t]$. This implies that the tail decreases very fast (geometrically fast) when $t > m$. Suppose $T + 2 > m$, then $\Pr[S_1 \ge T + u] \le (\frac{m}{T + 2})^{u-1} \Pr[Z \ge T + 1]$. So $\sum_{u \ge 1} \Pr[Z \ge T + u] \le \frac{\Pr[Z \ge T + 1]}{1 - m/(T + 2)} \le \Pr[Z \ge T] \cdot \frac{m}{T + 1} \cdot \frac{1}{1 - m / (T + 2)}$.

So by approximation, $\E[M] \lesssim T + k \Pr[Z \ge T] \cdot \frac{m}{T + 1} \cdot \frac{1}{1 - m / (T + 2)}$. We can pick $T$ such that $k \Pr[Z \ge T] \simeq 1$, then this would be a good upper bound. To find the $T$, we use the fact that $\Pr[Z \ge T] \simeq \exp(-m h(T/m))$ where $h(x) = x \ln x - x + 1$. Solving the equation we have $m h(T/m) \simeq \ln k$. Since $\ln k / m = \Theta(\ln \ln k)$ and $h(x) \rightarrow x \ln x$ when $x \rightarrow \infty$, so

$$
    \frac{T}{m} \ln (\frac{T}{m}) \simeq \Theta(\log \log k)
$$

So we can represent $T$ with Lambert $W$ function:

$$
\frac{T}{m} = \Theta\left(\frac{\log \log k}{W(\log \log k)}\right)
$$

Thus we have:

$$
T = \Theta\left(\frac{\log k}{W(\log \log k)}\right) = \Theta\left(\frac{\log k}{\log \log \log k}\right)
$$

So we know $\E[M] = \Theta\left(\frac{\log k}{\log \log \log k}\right)$ which means the a posteriori optimal cost is bounded by $k \cdot \Theta\left(\frac{\log k}{\log \log \log k}\right)$.

\section{Adaptivity gap lower bound: cycle-subdivision instance}

Our main result in this section is a lower bound of $5/4$ on the adaptivity gap for symmetric stochastic TSP. We show that there exists a family of instances obtained by subdividing the edges of a cycle in a complete graph for which the ratio $\optprior/\optadapt$ tends to $5/4$ as the number of vertices grows. Since $\optpost \le \optadapt$, the same lower bound applies to the obliviousness gap $\optprior/\optpost$. 

\begin{theorem}[Cycle-subdivision gap]\label{thm:cycle-subdivision}
The adaptivity gap for symmetric stochastic TSP is at least $\frac54$.
Since $\optpost\le\optadapt$ (cf.\ \eqref{eq:optineq}), the same lower bound holds for the obliviousness gap.
\end{theorem}

We prove this by exhibiting a family of instances and establishing a matching pair of bounds. Consider the following construction.

\begin{definition}[Cycle-subdivision instance]\label{def:cycle-sub}
Let $n\ge 7$ be odd. Start with the complete graph on vertices $v_1,\dots,v_n$, with every edge of length $1$, and fix the Hamiltonian cycle
\[
v_1v_2\cdots v_nv_1,
\]
where indices are taken modulo $n$. For each $i\in[n]$, replace the cycle edge $v_iv_{i+1}$ by a path $v_i v_i' v_{i+1}$ of two unit edges; let $d$ be the shortest-path metric of the resulting graph. Let $r:=v_1$ be the depot, and let the client set be
\[
V:=\{v_2,\dots,v_n,v_1',\dots,v_n'\}.
\]
Every original client $v_2,\dots,v_n$ is active with probability $1$, and every subdividing vertex $v_i'$ is active independently with probability $1/2$.
\end{definition}

Theorem~\ref{thm:cycle-subdivision} follows from the next two lemmas.

\begin{lemma}\label{lem:cycle-prior}
For every odd $n\ge 7$, the cycle-subdivision instance satisfies $\optprior=\frac{15}{8}n$.
\end{lemma}

\begin{lemma}\label{lem:cycle-adapt}
For every odd $n\ge 7$, the cycle-subdivision instance satisfies $\optadapt\le \frac32 n+6$.
\end{lemma}

\begin{proof}[Proof of Theorem~\ref{thm:cycle-subdivision}]
By Lemma~\ref{lem:cycle-prior}, $\optprior=\frac{15}{8}n$, and by Lemma~\ref{lem:cycle-adapt}, $\optadapt\le \frac32 n+6$.
Hence $\optprior/\optadapt$ converges to $\frac{5}{4}$ as $n$ grows.
Since $\optpost\le\optadapt$, the same lower bound holds for the obliviousness gap.
\end{proof}

We now prove the two lemmas in turn.

\begin{proof}[Proof of Lemma~\ref{lem:cycle-prior}]
Fix an arbitrary a priori tour $T=(r,s_1,s_2,\dots,s_{2n-1},r)$ on $\{r\}\cup V$, where $s_1,\dots,s_{2n-1}$ is a permutation of $\{v_2,\dots,v_n,v_1',\dots,v_n'\}$. We shall prove that its expected realized length is at least $15n/8$, and then exhibit a tour attaining this value.

\medskip
\noindent\textbf{1. Metric structure.}
For later use we record the relevant distances in the metric completion.
For original vertices,
\[
d(v_i,v_{i+1})=2,
\qquad
 d(v_i,v_j)=1 \quad \text{if } j\not\equiv i\pm 1 \pmod n.
\]
For subdividing vertices,
\[
d(v_i',v_i)=d(v_i',v_{i+1})=1,
\qquad
 d(v_i',v_j)=2 \quad \text{if } j\notin\{i,i+1\}.
\]
Finally, for $i\neq j$,
\[
d(v_i',v_j')=
\begin{cases}
2, & \text{if the cycle edges } v_iv_{i+1} \text{ and } v_jv_{j+1} \text{ are adjacent},\\
3, & \text{otherwise.}
\end{cases}
\]

\medskip
\noindent\textbf{2. Decomposition into segments.}
Deleting all subdividing vertices from $T$ leaves a cyclic ordering
\[
(r=u_1,u_2,\dots,u_n,r)
\]
of the original vertices $v_1,\dots,v_n$. For each $j\in[n]$, let
\[
W_j=(w_{j,1},\dots,w_{j,k_j})
\]
be the ordered list (possibly empty) of subdividing vertices appearing on $T$ between $u_j$ and $u_{j+1}$, where $u_{n+1}:=r$. Then
\[
\sum_{j=1}^n k_j=n,
\]
since every subdividing vertex appears in exactly one such segment.

Given original vertices $u,v$ and an ordered sequence $W=(w_1,\dots,w_k)$ of distinct subdividing vertices, write
\[
\Phi(u;W;v)
\]
for the expected length of the realized path obtained from the segment
\[
(u,w_1,\dots,w_k,v)
\]
after deleting inactive subdividing vertices. Because all original vertices are always active, the realized tour decomposes into these segment contributions, and therefore
\[
\E[\cost(T|_A)] = \sum_{j=1}^n \Phi(u_j;W_j;u_{j+1}).
\]
Thus it is enough to derive a lower bound on $\Phi(u;W;v)$ in terms of $k:=|W|$.

\medskip
\noindent\textbf{3. Short segments.}
We first treat the cases $k\le 3$.

\begin{claim}
For every segment $(u;W;v)$ with $k=|W|\le 3$,
\[
\Phi(u;W;v)\ge
\begin{cases}
1, & k=0,\\
2, & k=1,\\
\dfrac{11}{4}, & k=2,\\
\dfrac{29}{8}, & k=3.
\end{cases}
\]
Moreover, equality is attained for $k=2$ by $(v_i,v_i',v_{i+1}',v_{i+1})$, and for $k=3$ by $(v_{i+1},v_{i+1}',v_i',v_{i-1}',v_i)$.
\end{claim}

\begin{proof}
If $k=0$, then the segment contributes the single edge $uv$, so $\Phi(u;\varnothing;v)=d(u,v)\ge 1$.

If $k=1$, say $W=(w)$, then
\[
\Phi(u;w;v)=\tfrac12 d(u,v)+\tfrac12\bigl(d(u,w)+d(w,v)\bigr).
\]
If $u$ and $v$ are consecutive on the original cycle, then $d(u,v)=2$ and also $d(u,w)+d(w,v)\ge 2$. If they are not consecutive, then $d(u,v)=1$, and no subdividing vertex is incident with both $u$ and $v$, so $d(u,w)+d(w,v)\ge 3$. In either case $\Phi(u;w;v)\ge 2$.

Now let $k=2$, say $W=(w_1,w_2)$. There are four realizations.
If $d(u,v)=1$, then neither $w_1$ nor $w_2$ is incident with both $u$ and $v$, so each singleton realization has cost at least $3$, while the realization with both active has cost at least $4$. Hence
\[
\Phi(u;W;v)\ge \frac{1+3+3+4}{4}=\frac{11}{4}.
\]
If $d(u,v)=2$, then exactly one subdividing vertex is incident with both $u$ and $v$, so at most one singleton realization has cost $2$; the two singleton costs therefore sum to at least $5$, and the realization with both active has cost at least $4$. Thus
\[
\Phi(u;W;v)\ge \frac{2+5+4}{4}=\frac{11}{4}.
\]
Equality is attained, for example, by $(v_i,v_i',v_{i+1}',v_{i+1})$, whose four realization costs are $2,2,3,4$.

Finally let $k=3$, say $W=(w_1,w_2,w_3)$.
Suppose first that $d(u,v)=1$. Then each singleton realization has cost at least $3$. A two-vertex realization can have cost $4$ only if one active subdivider is incident with $u$, the other is incident with $v$, and the corresponding cycle edges are adjacent. For fixed nonconsecutive $u$ and $v$, there is at most one unordered pair of subdividing vertices with this property. Hence among the three two-vertex realizations, at most one has cost $4$, and the other two have cost at least $5$. The realization with all three active has cost at least $6$. Therefore
\[
\Phi(u;W;v)\ge \frac{1+9+14+6}{8}=\frac{15}{4}>\frac{29}{8}.
\]

Now suppose that $d(u,v)=2$. Then exactly one subdividing vertex is incident with both $u$ and $v$, so among the three singleton realizations at most one has cost $2$. Hence the sum of the singleton costs is at least $2+3+3=8$. A two-vertex realization can have cost $4$ only if one active subdivider is the unique subdivider of the edge $uv$ and the other comes from one of the two cycle edges adjacent to $uv$. Thus among the three two-vertex realizations, at most two have cost $4$, and the remaining one has cost at least $5$; the sum of these three costs is therefore at least $13$. The realization with all three active has cost at least $6$. Consequently,
\[
\Phi(u;W;v)\ge \frac{2+8+13+6}{8}=\frac{29}{8}.
\]
Equality is attained, for example, by $(v_{i+1},v_{i+1}',v_i',v_{i-1}',v_i)$, whose eight realization costs are
\[
2,\ 3,\ 2,\ 3,\ 4,\ 5,\ 4,\ 6.
\]
\end{proof}

\medskip
\noindent\textbf{4. Long segments.}
We now consider a segment
\[
(u;w_1,\dots,w_k;v)
\]
with $k\ge 4$. Set
\[
z_0:=u,\qquad z_t:=w_t\ (1\le t\le k),\qquad z_{k+1}:=v.
\]
Two retained vertices $z_i,z_j$ become consecutive after shortcutting precisely when every vertex strictly between them is deleted. Since each $w_t$ is retained independently with probability $1/2$, we obtain the exact expansion
\begin{align*}
\Phi(u;W;v)
&=2^{-k}d(u,v)
 +\sum_{t=1}^k 2^{-t}d(u,w_t)
 +\sum_{t=1}^k 2^{-(k-t+1)}d(w_t,v) \\
&\qquad
 +\sum_{1\le i<j\le k}2^{-(j-i+1)}d(w_i,w_j).
\end{align*}
Introduce the indicators
\[
\alpha_t:=\1_{\{w_t \text{ is incident with } u\}},
\qquad
\beta_t:=\1_{\{w_t \text{ is incident with } v\}},
\]
and, for $1\le i<j\le k$,
\[
\gamma_{ij}:=\1_{\{w_i,w_j \text{ correspond to adjacent cycle edges}\}}.
\]
By the distance formulas above,
\begin{align*}
d(u,v)&=1+\1_{\{u,v \text{ are consecutive on the original cycle}\}},\\
d(u,w_t)&=2-\alpha_t,\\
d(w_t,v)&=2-\beta_t,\\
d(w_i,w_j)&=3-\gamma_{ij}.
\end{align*}
Substituting these identities into the preceding formula yields
\[
\Phi(u;W;v)=1+\frac{3k}{2}+2^{-k}\1_{\{u,v\text{ consecutive}\}}-D,
\]
where
\[
D:=\sum_{t=1}^k 2^{-t}\alpha_t
 +\sum_{t=1}^k 2^{-(k-t+1)}\beta_t
 +\sum_{1\le i<j\le k}2^{-(j-i+1)}\gamma_{ij}.
\]
We now estimate $D$.
Since an original vertex is incident with exactly two subdividing vertices,
\[
\sum_{t=1}^k 2^{-t}\alpha_t\le 2^{-1}+2^{-2}=\frac34,
\qquad
\sum_{t=1}^k 2^{-(k-t+1)}\beta_t\le \frac34.
\]
For the last sum, consider the graph on vertex set $\{w_1,\dots,w_k\}$ in which two vertices are adjacent when the corresponding subdividing vertices arise from adjacent edges of the original cycle. This graph is a subgraph of a cycle, hence has maximum degree at most $2$ and therefore at most $k$ edges. Every edge contributes weight at most $1/4$, so
\[
\sum_{1\le i<j\le k}2^{-(j-i+1)}\gamma_{ij}\le \frac{k}{4}.
\]
Therefore
\[
D\le \frac34+\frac34+\frac{k}{4}=\frac32+\frac{k}{4},
\]
and hence
\[
\Phi(u;W;v)\ge 1+\frac{3k}{2}-\left(\frac32+\frac{k}{4}\right)=\frac54 k-\frac12.
\]
Since $k\ge 4$,
\[
\frac54 k-\frac12\ge 1+\frac{7k}{8},
\]
so every segment with $k\ge 4$ satisfies
\[
\Phi(u;W;v)\ge 1+\frac{7k}{8}.
\]
Combining this with the previous claim, we conclude that every segment with $k$ subdividing vertices satisfies
\[
\Phi(u;W;v)\ge
\begin{cases}
1, & k=0,\\
1+\dfrac{7k}{8}, & k\ge 1.
\end{cases}
\]

\medskip
\noindent\textbf{5. Global lower bound.}
Applying the segment bound to the decomposition of $T$, we obtain
\begin{align*}
\E[\cost(T|_A)]
&=\sum_{j=1}^n \Phi(u_j;W_j;u_{j+1})\\
&\ge \sum_{j:k_j=0} 1+\sum_{j:k_j\ge 1}\left(1+\frac{7k_j}{8}\right)\\
&=n+\frac78\sum_{j=1}^n k_j\\
&=n+\frac78 n = \frac{15}{8}n.
\end{align*}
Since $T$ was arbitrary, this shows that
\[
\optprior\ge \frac{15}{8}n.
\]

\medskip
\noindent\textbf{6. Matching constructions.}
It remains to exhibit a priori tours attaining this value.

\smallskip
\noindent\emph{Even $n$.}
Write $n=2m$. Consider the tour
\begin{align*}
T^{\mathrm{even}}
:=(&v_1,v_1',v_2',v_2,
   v_4,v_4',v_3',v_3,
   v_6,v_6',v_5',v_5,
   \dots,\\
  &v_{2m},v_{2m}',v_{2m-1}',v_{2m-1},v_1).
\end{align*}
This tour has $m$ nonempty segments, namely
\[
(v_1,v_1',v_2',v_2)
\quad\text{and}\quad
(v_{2t},v_{2t}',v_{2t-1}',v_{2t-1})\qquad (2\le t\le m),
\]
each with expected contribution $11/4$, together with $m$ direct edges between original vertices, each of length $1$. Hence
\[
\E[\cost(T^{\mathrm{even}}|_A)] = m\left(\frac{11}{4}+1\right)=\frac{15}{8}n.
\]

\smallskip
\noindent\emph{Odd $n$.}
Write $n=2m+1$ with $m\ge 2$. Consider the tour
\begin{align*}
T^{\mathrm{odd}}
:=(&v_n,v_2,v_2',v_1',v_n',v_1,
   v_4,v_4',v_3',v_3,
   v_6,v_6',v_5',v_5,
   \dots,\\
  &v_{2m},v_{2m}',v_{2m-1}',v_{2m-1},v_n).
\end{align*}
This tour has one $3$-segment,
\[
(v_2,v_2',v_1',v_n',v_1),
\]
with expected contribution $29/8$, together with $m-1$ segments
\[
(v_{2t},v_{2t}',v_{2t-1}',v_{2t-1})\qquad (2\le t\le m),
\]
each contributing $11/4$, and $m+1$ direct edges between original vertices, each of length $1$. Therefore
\begin{align*}
\E[\cost(T^{\mathrm{odd}}|_A)]
&=\frac{29}{8}+(m-1)\cdot \frac{11}{4}+(m+1)\\
&=\frac{15}{8}(2m+1)=\frac{15}{8}n.
\end{align*}

In both parity cases we have constructed an a priori tour of expected cost $15n/8$. Combined with the lower bound, this proves
\[
\optprior=\frac{15}{8}n.
\]
\end{proof}

\begin{proof}[Proof of Lemma~\ref{lem:cycle-adapt}]
We exhibit an adaptive policy $\pi$.
Let $P:=v_2,v_2',v_3,\dots,v_{n-1}',v_n$. First call $v_1'$ and $v_n'$ (the only clients outside~$P$), then return to the depot $r=v_1$. In the worst case both are active, costing $d(r,v_1')+d(v_1',v_n')+d(v_n',r)=1+2+1=4$.
Next, move from $r$ to $v_2$ (cost~$2$) and process~$P$.
At each step the uncalled clients on~$P$ form a contiguous subpath, and the salesperson sits at one of its two endpoint originals. The policy probes the adjacent subdividing vertex: if it is active, traverse it and the next original (cost~$2$) and continue in the same direction; if it is inactive, jump to the opposite endpoint original of the remaining subpath (cost~$1$, since nonconsecutive originals are at distance~$1$) and proceed symmetrically.

Let $a:=|A\cap\{v_2',\dots,v_{n-1}'\}|$. The $a$ active subdividers on~$P$ each contribute cost~$2$. Each of the $(n-2)-a$ inactive ones triggers a switch of cost~$1$, except possibly the last, which may cost~$2$. Including the return to~$r$ (cost at most~$2$), the middle phase costs at most $2+2a+((n-2)-a+1)+2=n+a+3$.
The total cost per realization is therefore at most $4+(n+a+3)=n+a+7$, and taking expectation with $\E[a]=(n-2)/2$ gives $\E[\cost(\pi)]\le \frac32 n+6$.
\end{proof}

\section{Clairvoyance gap lower bound: asymmetric recursive instance}\label{sec:asym-recursive}

We now turn to asymmetric (directed) metrics and exhibit a family of instances
for which the clairvoyance gap $\optadapt/\optpost$ tends to~$4/3$.

\subsection*{The construction}

We define a family of asymmetric stochastic TSP instances.  Distances are taken to be the directed shortest-path metric induced by the directed edges described below.  The depot is always present.  The client vertices are independent, with probabilities specified below.

For every integer $l\geq 0$, define a directed recursive block $B_l$.  Each block has a distinguished set of permanent vertices, denoted $P_l$.  These permanent vertices will be deterministic clients.  For $l=0$, the block $B_0$ consists of a single permanent vertex.

For $l\geq 1$, the block $B_l$ is obtained from two disjoint copies of $B_{l-1}$, denoted $B_{l-1}^{\mathrm L}$ and $B_{l-1}^{\mathrm R}$, together with a new midpoint vertex $m_l$.  Let
\[
    w_l := 2^{l-1}.
\]
For every permanent vertex $\ell\in P(B_{l-1}^{\mathrm L})$ and every permanent vertex $r\in P(B_{l-1}^{\mathrm R})$, add a directed edge
\[
    \ell \to r
\]
of length $w_l$.  For every $r\in P(B_{l-1}^{\mathrm R})$, add a directed edge
\[
    r\to m_l
\]
of length $w_l$.  For every $\ell\in P(B_{l-1}^{\mathrm L})$, add a directed edge
\[
    m_l\to \ell
\]
of length $w_l$.  Thus, after contracting the two child blocks, the top level of $B_l$ is the directed triangle
\[
    B_{l-1}^{\mathrm L}
    \longrightarrow
    B_{l-1}^{\mathrm R}
    \longrightarrow
    m_l
    \longrightarrow
    B_{l-1}^{\mathrm L},
\]
where each displayed arc has length $w_l$.

The permanent set of $B_l$ is
\[
    P_l := P(B_{l-1}^{\mathrm L})\cup P(B_{l-1}^{\mathrm R}).
\]
Thus $|P_l|=2^l$.  All permanent vertices are active with probability $1$.  Every midpoint vertex is active independently with probability $1/2$.

The full rooted instance $I_l$ is obtained by adding a depot $r$.  For every permanent vertex $\ell\in P_l$, add directed edges
\[
    r\to \ell,
    \qquad
    \ell\to r,
\]
each of length $w_l=2^{l-1}$.

Let $M_l$ be the set of midpoint vertices in $B_l$.  A midpoint of local depth $k$ has weight $w_k=2^{k-1}$.  There are $2^{l-k}$ midpoint vertices of local depth $k$ inside $B_l$.

\subsection*{Statement}

\begin{theorem}\label{thm:recursive-gap}
For every $l\geq 1$, the instance $I_l$ satisfies
\[
    \optpost(I_l)=(3l+4)2^{l-2},
\]
and
\[
    \optadapt(I_l)=(l+1)2^l.
\]
Consequently,
\[
    \frac{\optadapt(I_l)}{\optpost(I_l)}
    =
    \frac{4(l+1)}{3l+4}
    \longrightarrow
    \frac43.
\]
\end{theorem}

The rest of this section proves the theorem.

\subsection*{A posteriori cost}

We first analyze an open version of the block.  For a realization $A\subseteq M_l$ of the active midpoint vertices, let $P_l(A)$ be the minimum internal cost of a directed path that serves all deterministic permanent vertices of $B_l$ and all midpoint vertices in $A$, with free choice of the first and last permanent vertices of $B_l$.  The cost of entering the block from outside and leaving the block is not included in $P_l(A)$.

For $l=0$, there is only one deterministic permanent vertex, so
\[
    P_0(A)=0.
\]

\begin{lemma}\label{lem:post-rec}
For $l\geq 1$,
\[
    P_l(A)
    =
    P_{l-1}(A_{\mathrm L})
    +
    P_{l-1}(A_{\mathrm R})
    +
    w_l\bigl(1+\1_{\{m_l\in A\}}\bigr),
\]
where $A_{\mathrm L}$ and $A_{\mathrm R}$ are the active midpoint sets inside the left and right child blocks.
\end{lemma}

\begin{proof}
All permanent vertices are deterministic, so any feasible path inside $B_l$ must serve both child blocks $B_{l-1}^{\mathrm L}$ and $B_{l-1}^{\mathrm R}$.  By induction, the internal cost required inside the two child blocks is at least
\[
    P_{l-1}(A_{\mathrm L})+P_{l-1}(A_{\mathrm R}).
\]
It remains to determine the minimum top-level cost needed to move among the two child blocks and the midpoint $m_l$.

If $m_l\notin A$, then the path only needs to pass between the two child blocks.  The cheapest such top-level transition is
\[
    B_{l-1}^{\mathrm L}\to B_{l-1}^{\mathrm R},
\]
which costs $w_l$.  Therefore the top-level contribution is at least $w_l$, and this is achievable.

If $m_l\in A$, then the path must visit all three top-level objects
\[
    B_{l-1}^{\mathrm L},
    \qquad
    B_{l-1}^{\mathrm R},
    \qquad
    m_l.
\]
Since the path starts and ends at permanent vertices of $B_l$, it starts and ends in one of the two child blocks.  The cheapest top-level order is
\[
    B_{l-1}^{\mathrm R}\to m_l\to B_{l-1}^{\mathrm L},
\]
which costs $2w_l$.  No smaller top-level cost is possible, because visiting the midpoint and both child blocks requires at least two nonzero top-level transitions, and every such transition has length at least $w_l$ in the directed shortest-path metric.  Hence the top-level contribution is exactly $2w_l$.

Combining the child-block costs and the top-level cost gives the recurrence.
\end{proof}

Taking expectation in Lemma~\ref{lem:post-rec}, and using $\Pr[m_l\in A]=1/2$, gives
\[
    \E[P_l(A)]
    =
    2\E[P_{l-1}(A)]+\frac32 w_l.
\]
Equivalently, by iterating over all midpoint vertices,
\[
    P_l(A)
    =
    \sum_{x\in M_l}w(x)
    +
    \sum_{x\in A}w(x),
\]
where $w(x)$ denotes the local weight of midpoint $x$.  Hence
\[
    \E[P_l(A)]
    =
    \frac32\sum_{x\in M_l}w(x).
\]
Now
\[
    \sum_{x\in M_l}w(x)
    =
    \sum_{k=1}^l 2^{l-k}2^{k-1}
    =
    \sum_{k=1}^l 2^{l-1}
    =
    l2^{l-1}.
\]
Therefore
\[
    \E[P_l(A)]
    =
    \frac32 l2^{l-1}
    =
    3l2^{l-2}.
\]

In the rooted instance $I_l$, the a posteriori tour must enter the block from the depot and return to the depot.  The cheapest edge from the depot to the block has length $w_l$, and the cheapest edge from the block back to the depot also has length $w_l$.  Thus the depot contribution is
\[
    2w_l=2^l.
\]
It follows that
\[
    \optpost(I_l)
    =
    2^l+3l2^{l-2}
    =
    (3l+4)2^{l-2}.
\]

\subsection*{Adaptive cost}

We now analyze the optimal adaptive policy.  Again we first consider an open block.  Let $Q_l$ be the minimum expected internal cost of an adaptive policy for $B_l$, where the policy may choose the first permanent vertex of $B_l$ for free and, after serving/calling all clients in the block, may exit for free from any permanent vertex of $B_l$.  In particular, if the policy ends at a midpoint, it must still pay the distance from that midpoint to a permanent vertex before exiting the block.

Clearly
\[
    Q_0=0.
\]

The following elementary three-state lemma is the key point.

\begin{lemma}\label{lem:three-state}
Consider three top-level objects $L,R,M$.  The objects $L$ and $R$ are deterministic, while $M$ is active independently with probability $1/2$.  The directed distances are the shortest-path metric induced by the directed cycle
\[
    L\to R\to M\to L,
\]
where every displayed arc has length $w$.  The policy starts for free at either $L$ or $R$, with that object already served, and must serve the other deterministic object and call $M$.  If $M$ is active, it must be visited immediately when called.  The policy may exit for free from $L$ or $R$ after all required calls have been made.  Then the minimum expected cost is exactly $2w$.
\end{lemma}

\begin{proof}
By scaling, assume $w=1$.

First, there is a policy of expected cost $2$.  Start at $L$, serve $R$, and then call $M$.  The move $L\to R$ costs $1$.  If $M$ is inactive, the policy is already at a permanent point and exits.  If $M$ is active, the policy travels $R\to M$, costing $1$, and then returns from $M$ to a permanent point, costing $1$ via $M\to L$.  Thus the expected cost is
\[
    1+\frac12(1+1)=2.
\]

It remains to prove the lower bound.  Suppose first that the policy starts at $L$.  If it serves $R$ before calling $M$, then it already pays $d(L,R)=1$.  If $M$ is active when called from $R$, it must pay $d(R,M)=1$ and then must return from $M$ to a permanent point, which costs at least $1$.  Hence this case has expected cost at least
\[
    1+\frac12(1+1)=2.
\]
If instead the policy calls $M$ before serving $R$, then if $M$ is inactive it must still move from $L$ to $R$, costing $1$.  If $M$ is active, it must move from $L$ to $M$, whose directed distance is $2$, and then still reach $R$ before exiting, whose directed distance from $M$ is $2$.  Hence this case has expected cost at least
\[
    \frac12\cdot 1+\frac12\cdot(2+2)>2.
\]
Thus every policy starting at $L$ has expected cost at least $2$.

Now suppose the policy starts at $R$.  If it calls $M$ before serving $L$, then if $M$ is active it pays $d(R,M)+d(M,L)=1+1=2$, while if $M$ is inactive it must still move from $R$ to $L$, whose directed distance is $2$.  Therefore the expected cost is at least $2$.  If it serves $L$ before calling $M$, then it already pays $d(R,L)=2$, and all further expected cost is nonnegative.  Hence every policy starting at $R$ also has expected cost at least $2$.

The lower bound and upper bound coincide.  Rescaling gives the value $2w$.
\end{proof}

\begin{lemma}\label{lem:adapt-rec}
For every $l\geq 1$,
\[
    Q_l=2Q_{l-1}+2w_l.
\]
\end{lemma}

\begin{proof}
Contract the left child block to $L$, the right child block to $R$, and the midpoint $m_l$ to $M$.  The top-level geometry is exactly the three-state instance of Lemma~\ref{lem:three-state}, with $w=w_l$.

For the upper bound, use an optimal policy from Lemma~\ref{lem:three-state} at the top level.  Whenever the policy enters one of the two child blocks, execute an optimal adaptive policy for that child block.  The two child blocks are independent copies of $B_{l-1}$, and the midpoint $m_l$ is independent of both of them.  This gives expected cost
\[
    2Q_{l-1}+2w_l.
\]

For the lower bound, consider any adaptive policy for $B_l$.  The restriction of the policy to each child block is an adaptive policy for that child block.  Since the random activations in the two child blocks are independent of all information outside the corresponding block, the expected internal cost incurred in each child block is at least $Q_{l-1}$.  Moreover, after contracting each child block to a top-level object, the projected policy must solve the three-state problem of Lemma~\ref{lem:three-state}.  Therefore its expected top-level cost is at least $2w_l$.  Hence every adaptive policy has expected cost at least
\[
    2Q_{l-1}+2w_l.
\]
The upper and lower bounds match.
\end{proof}

Using $Q_0=0$ and $w_l=2^{l-1}$, Lemma~\ref{lem:adapt-rec} gives
\[
    Q_l
    =
    2Q_{l-1}+2^l.
\]
Solving the recurrence,
\[
    Q_l
    =
    \sum_{k=1}^l 2^{l-k}2^k
    =
    \sum_{k=1}^l 2^l
    =
    l2^l.
\]

In the rooted instance $I_l$, the policy must enter the block from the depot and finally return to the depot.  These two movements contribute $2w_l=2^l$.  Therefore
\[
    \optadapt(I_l)
    =
    2^l+Q_l
    =
    2^l+l2^l
    =
    (l+1)2^l.
\]

\subsection*{The ratio}

Combining the two formulas,
\[
    \frac{\optadapt(I_l)}{\optpost(I_l)}
    =
    \frac{(l+1)2^l}{(3l+4)2^{l-2}}
    =
    \frac{4(l+1)}{3l+4}.
\]
Therefore
\[
    \lim_{l\to\infty}
    \frac{\optadapt(I_l)}{\optpost(I_l)}
    =
    \frac43.
\]
For example, when $l=3$,
\[
    \optpost(I_3)=26,
    \qquad
    \optadapt(I_3)=32,
\]
and hence
\[
    \frac{\optadapt(I_3)}{\optpost(I_3)}
    =
    \frac{32}{26}
    =
    \frac{16}{13}
    \approx 1.23077.
\]
For $l=4$, the formulas give
\[
    \optpost(I_4)=64,
    \qquad
    \optadapt(I_4)=80,
\]
so the ratio is $5/4$.

This completes the proof of Theorem~\ref{thm:recursive-gap}. \qed

\section{A survival-local packet DP for directed a priori TSP}\label{sec:survival-dp}

In this section we present a $(1+\varepsilon)$-approximation algorithm for the rooted a priori TSP on directed metrics, running in time $c_\varepsilon^n \poly(n)$.  The approach combines survival-window locality with a packet decomposition, resulting in a surrogate objective that can be minimized exactly by dynamic programming.

\subsection*{Setup}

Let
\[
    \pi=(r,v_1,\dots,v_n,r)
\]
be a rooted a priori permutation.  Put $v_0=v_{n+1}=r$ and
$p_{v_0}=p_{v_{n+1}}=1$.  The distance function $d$ is a directed metric.  For
notational brevity write
\[
    p_i:=p_{v_i},
    \qquad
    d_{ij}:=d(v_i,v_j).
\]
For a client $v_i$, define
\[
    \ell_i:=-\log(1-p_i)\in[0,\infty],
\]
with the convention $\ell_i=\infty$ when $p_i=1$.  For $i<j$, put
\[
    L(i,j):=\sum_{i<k<j}\ell_k,
    \qquad
    Q(i,j):=\e^{-L(i,j)}=\prod_{i<k<j}(1-p_k).
\]
Thus $Q(i,j)$ is exactly the probability that all vertices strictly between
$v_i$ and $v_j$ are inactive.  The a priori cost of $\pi$ is
\[
    \cost(\pi)
    =
    \sum_{0\le i<j\le n+1}w_{ij}d_{ij},
    \qquad
    w_{ij}:=p_i p_j Q(i,j).
\]

\begin{definition}[Survival-local cost]\label{def:survival-local}
For $\Gamma>0$, define
\[
    \cost^{\loc}_{\Gamma}(\pi)
    :=
    \sum_{\substack{0\le i<j\le n+1\\ L(i,j)<\Gamma}}
    w_{ij}d_{ij}.
\]
Equivalently, the retained terms are those for which
$Q(i,j)>\e^{-\Gamma}$.
\end{definition}

\subsection*{Survival-window locality}

We first prove that the full a priori cost is well approximated by its
survival-local part.  The proof uses only a local pivot window and deterministic
first-active weights.

\begin{lemma}[One-sided survival tails]\label{lem:survival-tail}
Let $\lambda>0$ and $\eta:=\e^{-\lambda}$.
\begin{enumerate}[label=(\roman*)]
\item For every fixed $z$,
\[
    \sum_{\substack{j>z\\ \ell_z+L(z,j)\ge\lambda}}
    p_j\e^{-(\ell_z+L(z,j))}
    \le \eta,
\]
where the terminal depot $j=n+1$ is included with $p_{n+1}=1$.

\item For every fixed $z$,
\[
    \sum_{\substack{i<z\\ L(i,z)+\ell_z\ge\lambda}}
    p_i\e^{-(L(i,z)+\ell_z)}
    \le \eta,
\]
where the initial depot $i=0$ is included with $p_0=1$.
\end{enumerate}
\end{lemma}

\begin{proof}
We prove (i); the proof of (ii) is the same in the reversed order.  Put
\[
    T_j:=\ell_z+L(z,j)=\sum_{z\le k<j}\ell_k.
\]
The set of indices $j$ with $T_j\ge\lambda$ is a suffix.  If $j_0$ is its first
index, then telescoping gives
\[
    \sum_{j\ge j_0} p_j\e^{-T_j}=\e^{-T_{j_0}},
\]
where the final depot term is included by $p_{n+1}=1$.  Since
$T_{j_0}\ge\lambda$, the sum is at most $\e^{-\lambda}$.
\end{proof}

\begin{lemma}[Survival-window locality]\label{lem:survival-window-locality}
Let $\lambda\ge1$, set $\eta:=\e^{-\lambda}$, and put
\[
    \Gamma:=3\lambda.
\]
Then every a priori permutation $\pi$ satisfies
\[
    \cost(\pi)
    \le
    \bigl(1+O(\eta)\bigr)\cost^{\loc}_{\Gamma}(\pi),
\]
where the hidden constant is universal.
\end{lemma}

\begin{proof}
Let
\[
    C_{\loc}:=\cost^{\loc}_{\Gamma}(\pi),
    \qquad
    H:=\cost(\pi)-\cost^{\loc}_{\Gamma}(\pi).
\]
We shall prove the stronger explicit inequality
\[
    H\le \frac{\eta}{1-\eta}(2C_{\loc}+H).
\]
Since $\lambda\ge1$, we have $\eta<1/2$, and this will imply
\[
    H\le \frac{2\eta}{1-2\eta}C_{\loc}=O(\eta)C_{\loc}.
\]

Fix a left endpoint $i$.  Define the pivot window
\[
    W_i
    :=
    \{z\in\{i+1,\dots,n\}: L(i,z)<2\lambda
       \text{ and } L(i,z)+\ell_z\ge\lambda\}.
\]
If $(i,j)$ is nonlocal, i.e.\ $L(i,j)\ge3\lambda$, then $W_i$ lies before $j$ and
has total log-survival length at least $\lambda$:
\[
    \sum_{z\in W_i}\ell_z\ge\lambda.
\]
Indeed, scanning rightward from $i$, the cumulative log-survival length must
cross the whole interval $[\lambda,2\lambda]$ before reaching $j$.

For $z\in W_i$, define the deterministic first-active weight
\[
    a_i(z)
    :=
    p_z\prod_{\substack{u\in W_i\\ u<_\pi z}}(1-p_u).
\]
Then
\[
    A_i:=\sum_{z\in W_i}a_i(z)
    =1-\prod_{z\in W_i}(1-p_z)
    \ge 1-\e^{-\lambda}=1-\eta.
\]
For every nonlocal pair $(i,j)$ and every $z\in W_i$, the directed triangle
inequality gives
\[
    d_{ij}\le d_{iz}+d_{zj}.
\]
Averaging this deterministic inequality with weights $a_i(z)/A_i$, multiplying
by $w_{ij}$, and summing over all nonlocal pairs gives
\[
    H\le \frac{1}{1-\eta}(S_1+S_2),
\]
where $S_1$ and $S_2$ are the resulting sums with distances $d_{iz}$ and
$d_{zj}$, respectively.

We bound $S_1$.  Fix $i$ and $z\in W_i$.  Since $L(i,z)<2\lambda<\Gamma$, the
pair $(i,z)$ is local.  For every nonlocal $j>z$,
\[
    L(i,j)=L(i,z)+\ell_z+L(z,j)\ge3\lambda,
\]
and therefore $\ell_z+L(z,j)\ge\lambda$.  Using the definition of $a_i(z)$ and
discarding a factor at most $1$,
\[
    \sum_{\substack{j>z\\L(i,j)\ge3\lambda}}
    w_{ij}a_i(z)
    \le
    w_{iz}
    \sum_{\substack{j>z\\\ell_z+L(z,j)\ge\lambda}}
    p_j\e^{-(\ell_z+L(z,j))}.
\]
By Lemma~\ref{lem:survival-tail}, the last sum is at most $\eta$.  Hence
\[
    S_1\le \eta C_{\loc}.
\]

We bound $S_2$.  Fix $z<j$.  For every $i$ with $z\in W_i$, we have
$L(i,z)+\ell_z\ge\lambda$.  Again, from the definition of $a_i(z)$ and by
discarding a factor at most $1$,
\[
    \sum_{\substack{i<z\\ z\in W_i\\ L(i,j)\ge3\lambda}}
    w_{ij}a_i(z)
    \le
    w_{zj}
    \sum_{\substack{i<z\\L(i,z)+\ell_z\ge\lambda}}
    p_i\e^{-(L(i,z)+\ell_z)}.
\]
By Lemma~\ref{lem:survival-tail}, the last sum is at most $\eta$.  Therefore
\[
    S_2\le \eta\cost(\pi)=\eta(C_{\loc}+H).
\]
Combining the two estimates,
\[
    H\le \frac{\eta}{1-\eta}(2C_{\loc}+H).
\]
As noted at the beginning of the proof, this gives $H\le O(\eta)C_{\loc}$.
Hence
\[
    \cost(\pi)=C_{\loc}+H
    \le
    \bigl(1+O(\eta)\bigr)\cost^{\loc}_{\Gamma}(\pi).
\]
\end{proof}

\begin{remark}\label{rmk:no-block}
No global block decomposition is used.  The weights $a_i(z)$ are not an
auxiliary random sample; they are deterministic first-active weights on the
pivot window and provide exactly the endpoint factor $p_z$ needed in the
coefficient comparison.
\end{remark}

\subsection*{Packets}

Fix $0<\delta<1/4$ and set $\theta:=\delta/2$.  A client is \emph{large} if
$p_v>\theta$ and \emph{small} otherwise.  A unit is either a large singleton or
a nonempty packet of small clients.

For a unit $U$, define
\[
    \mu(U):=\sum_{v\in U}p_v,
    \qquad
    q(U):=\prod_{v\in U}(1-p_v),
    \qquad
    \ell(U):=-\log q(U)=\sum_{v\in U}\ell(v).
\]

\begin{definition}[Admissibility]\label{def:admissible}
A packet sequence $\UU=(U_1,\dots,U_m)$ is $\delta$-admissible if it
partitions the client set, every large client is a singleton unit, every small
packet $B$ satisfies
\[
    \mu(B)\le\delta,
\]
and every small packet immediately preceded by another small packet satisfies
\[
    \mu(B)\ge\delta/2.
\]
Thus a tiny small packet is allowed only as the first packet of a maximal run of
small packets.
\end{definition}

\begin{lemma}[Consecutive admissible packetization]\label{lem:existence}
Every a priori permutation $\pi$ admits a consecutive $\delta$-admissible
packetization.
\end{lemma}

\begin{proof}
Keep each large client as a singleton.  This splits the remaining clients into
maximal runs of small clients.  In each run, scan from right to left, closing a
packet as soon as its mass reaches $\delta/2$.  Since each small client has
probability at most $\delta/2$, every closed packet has mass in
$[\delta/2,\delta]$.  The only possible leftover is the leftmost packet of the
run, and it may have mass below $\delta/2$.
\end{proof}

\subsection*{The packet surrogate}

For a small packet $B$ and an ordering $\sigma$ of $B$, define
\[
    P_B(\sigma)
    :=
    \sum_{x<_\sigma y}p_xp_y d(x,y),
    \qquad
    I(B):=\min_\sigma P_B(\sigma).
\]
For a large singleton and for the depot unit $\{r\}$, set $I=0$.  Fix one
minimizing order for every small packet.  Given a packet sequence $\UU$,
let $\pi(\UU)$ be the a priori permutation obtained by replacing each
small packet by its chosen minimizing order.

For units $U,W$, define
\[
    C(U,W):=\sum_{x\in U}\sum_{y\in W}p_xp_y d(x,y),
\]
where $p_r=1$ if one of the units is the depot.  Put $U_0=U_{m+1}=\{r\}$.  The
survival-local packet surrogate is
\[
\widetilde\cost_\Gamma(\UU)
:=
\sum_{i=1}^m I(U_i)
+
\sum_{\substack{0\le i<j\le m+1\\
        \sum_{i<t<j}\ell(U_t)<\Gamma}}
\left(\prod_{i<t<j}q(U_t)\right) C(U_i,U_j).
\]
The surrogate uses the same survival-local condition at the unit level and
drops the survival factors lying inside the two endpoint units.

\begin{lemma}[Upper envelope]\label{lem:envelope}
For every $\delta$-admissible packet sequence $\UU$,
\[
    \cost_\Gamma^{\loc}(\pi(\UU))
    \le
    \widetilde\cost_\Gamma(\UU).
\]
\end{lemma}

\begin{proof}
Consider a local exact term with endpoints $x\in U_i$ and $y\in U_j$, $i<j$.
Its intermediate survival product equals
\[
\left(\prod_{\substack{z\in U_i\\ z\text{ after }x}}(1-p_z)\right)
\left(\prod_{i<t<j}q(U_t)\right)
\left(\prod_{\substack{z\in U_j\\ z\text{ before }y}}(1-p_z)\right).
\]
Dropping the two endpoint factors can only increase the coefficient.  Moreover,
if the exact pair is $\Gamma$-local, then the complete intermediate units
between $U_i$ and $U_j$ have total $\ell$-length below $\Gamma$, so the
surrogate includes this inter-unit contribution.

If $x,y$ lie in the same small packet $B$, then the exact coefficient is at most
$p_xp_y$, and the chosen internal order contributes at most $I(B)$.  Large
singletons have no internal pairs.  Summing all terms proves the claim.
\end{proof}

\begin{lemma}[Comparison with a fixed order]\label{lem:comparison}
Let $\UU$ be a consecutive packetization of a a priori permutation $\pi$,
with every large client singleton and every small packet of mass at most
$\delta$.  Then
\[
    \widetilde\cost_\Gamma(\UU)
    \le
    \frac{1}{(1-\delta)^2}\cost(\pi).
\]
\end{lemma}

\begin{proof}
Every surrogate inter-unit term corresponds to an exact term of $\cost(\pi)$.
The only difference is that the surrogate omits survival factors inside the left
and right endpoint units.  Such a factor is $1$ for a depot or large singleton,
and is at least $1-\delta$ for a small packet, by the union bound.  Hence every
inter-unit surrogate coefficient is at most $(1-\delta)^{-2}$ times the
corresponding exact coefficient.

For a small packet $B$, let $\sigma_\pi$ be its order inside $\pi$, and let
$T_B(\sigma_\pi)$ be its exact internal contribution to $\cost(\pi)$.  Since
$\mu(B)\le\delta$, every internal survival factor is at least $1-\delta$.  Thus
\[
    I(B)
    \le P_B(\sigma_\pi)
    \le \frac{1}{1-\delta}T_B(\sigma_\pi)
    \le \frac{1}{(1-\delta)^2}T_B(\sigma_\pi).
\]
Summing the inter-unit and internal comparisons proves the lemma.
\end{proof}

\subsection*{Exact minimization of the surrogate}

The internal value $I(B)$ of each small candidate packet $B$ with
$\mu(B)\le\delta$ is precomputed by
\[
    I(B)=\min_{y\in B}\left(I(B\setminus\{y\})+
    \sum_{x\in B\setminus\{y\}}p_xp_y d(x,y)\right),
    \qquad I(\emptyset)=0.
\]
This is the usual last-vertex recurrence.

A prefix state consists of the placed set $S\subseteq V$ and the remembered
suffix $\Omega$ of the current unit sequence, where the initial depot is treated
as the first unit.  If the whole prefix has client $\ell$-length below $\Gamma$,
then $\Omega$ is the whole prefix.  Otherwise $\Omega$ is the shortest suffix
whose total client $\ell$-length is at least $\Gamma$.

A feasible next unit is either an unplaced large singleton, or a nonempty subset
of unplaced small clients of mass at most $\delta$.  If the current last unit is
a small packet, then a new small packet must have mass at least $\delta/2$.  When
a feasible unit $U$ is appended, the DP adds
\[
    I(U)+
    \sum_{W\in\Omega:\,\ell(W,U)<\Gamma}
    \e^{-\ell(W,U)}C(W,U),
\]
where $\ell(W,U)$ is the total $\ell$-length of the complete units strictly
between $W$ and the new unit $U$ in the current prefix.  After the transition,
the remembered suffix is trimmed by the same rule.  At the end, the terminal
depot is appended once, with internal cost zero.

\begin{lemma}[DP exactness]\label{lem:dp-exact}
The dynamic program returns a $\delta$-admissible packet sequence minimizing
$\widetilde\cost_\Gamma$.
\end{lemma}

\begin{proof}
When a new unit $U$ is appended, the only newly determined surrogate terms are
$I(U)$ and the inter-unit terms whose right endpoint is $U$.  Such a term from a
previous unit $W$ is present exactly when the complete units between $W$ and
$U$ have total $\ell$-length below $\Gamma$, and then its coefficient is
$\e^{-\ell(W,U)}C(W,U)$.

A forgotten unit has, by definition of the remembered suffix, at least $\Gamma$
client $\ell$-length between itself and the current end.  Future append
operations only increase this intervening length.  Hence a forgotten unit can
never again participate in a surrogate term.  Therefore the state contains
precisely the history relevant to all future costs, and the recurrence is an
exact shortest-path computation over all admissible prefixes.  The terminal
depot is handled by the same argument.
\end{proof}

\begin{lemma}[State count]\label{lem:state-count}
For fixed $\delta$ and $\Gamma$, the DP has running time
$c_{\delta,\Gamma}^n\poly(n)$.
\end{lemma}

\begin{proof}
The remembered suffix has total activation mass below $\Gamma+1$.  If it is the
whole prefix, this follows from $p_v\le\ell(v)$.  Otherwise, removing its first
unit leaves client $\ell$-length below $\Gamma$, hence activation mass below
$\Gamma$, and the first unit has activation mass at most $1$.

Every remembered large singleton has mass greater than $\delta/2$.  Every
remembered small packet except the first packet of a small run has mass at
least $\delta/2$.  A tiny first packet of a small run can be charged to the
preceding large singleton, except possibly for the first remembered unit.  Thus
the number of remembered units is
\[
    K=O\!\left(\frac{\Gamma+1}{\delta}\right).
\]
A state is encoded by assigning each client one of the labels ``unplaced'',
``placed and forgotten'', or ``remembered position $1,\dots,K$''.  This gives at
most $(K+2)^n$ states.  Including choices of the next unit gives at most
$(K+3)^n$ transition incidences up to polynomial factors.  Since $K$ depends
only on $\delta$ and $\Gamma$, the claimed bound follows.
\end{proof}

\subsection*{Approximation theorem}

\begin{theorem}\label{thm:survival-dp}
For every fixed $0<\varepsilon\le1$, the rooted a priori TSP on directed metrics
has a deterministic algorithm running in time $c_\varepsilon^n\poly(n)$ and
returning a a priori permutation $\hat\pi$ with
\[
    \cost(\hat\pi)\le(1+\varepsilon)\opt.
\]
\end{theorem}

\begin{proof}
Let $c_0$ be the universal constant implicit in
Lemma~\ref{lem:survival-window-locality}; thus
\[
    \cost(\pi)\le(1+c_0\e^{-\lambda})\cost_\Gamma^{\loc}(\pi)
    \qquad(\Gamma=3\lambda).
\]
Choose
\[
    \delta:=\varepsilon/16,
    \qquad
    \lambda:=\log(4c_0/\varepsilon),
    \qquad
    \Gamma:=3\lambda.
\]
Let $\pi^*$ be an optimum a priori permutation.  By
Lemma~\ref{lem:existence}, $\pi^*$ has a consecutive $\delta$-admissible
packetization $\UU^*$.  By Lemma~\ref{lem:comparison},
\[
    \widetilde\cost_\Gamma(\UU^*)
    \le
    (1-\delta)^{-2}\opt.
\]
Let $\widehat{\UU}$ be the DP minimizer and put
$\hat\pi:=\pi(\widehat{\UU})$.  By DP optimality and
Lemma~\ref{lem:envelope},
\[
    \cost_\Gamma^{\loc}(\hat\pi)
    \le
    \widetilde\cost_\Gamma(\widehat{\UU})
    \le
    \widetilde\cost_\Gamma(\UU^*)
    \le
    (1-\delta)^{-2}\opt.
\]
Using survival-window locality,
\[
    \cost(\hat\pi)
    \le
    (1+c_0\e^{-\lambda})(1-\delta)^{-2}\opt.
\]
By the choice of $\lambda$, $c_0\e^{-\lambda}\le\varepsilon/4$.  Also, for
$0<\varepsilon\le1$,
\[
    (1-\delta)^{-2}\le 1+4\delta\le 1+\varepsilon/4.
\]
Therefore
\[
    \cost(\hat\pi)
    \le
    (1+\varepsilon/4)^2\opt
    \le
    (1+\varepsilon)\opt.
\]
The running time follows from Lemma~\ref{lem:state-count}, since $\delta$ and
$\Gamma$ depend only on $\varepsilon$.
\end{proof}
