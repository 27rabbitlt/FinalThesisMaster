% --- Abstract ---
\chapter*{Abstract}
\addcontentsline{toc}{chapter}{Abstract}

We study the stochastic traveling salesperson problem, in which each vertex of a metric space is active independently with a given probability and the goal is to find a tour of minimum expected cost visiting all active vertices.
Three benchmarks of increasing information are compared: the \emph{a~posteriori} optimum (full knowledge of the active set), the \emph{adaptive} optimum (vertices are probed during the tour), and the \emph{a~priori} optimum (a fixed tour with shortcutting).

We prove several quantitative separations between these benchmarks.
First, we construct a family of symmetric instances showing that the adaptivity gap (a~priori vs.\ adaptive) is at least~$5/4$.
Second, we exhibit a family of symmetric instances achieving a clairvoyance gap (adaptive vs.\ a~posteriori) of~$11/10$, and prove partial upper bounds for the corresponding symmetric upper-bound conjecture.
Third, for asymmetric metrics, we construct a recursive family of instances for which the clairvoyance gap tends to~$4/3$.

Additionally, we present a $(1{+}\varepsilon)$-approximation algorithm for a~priori TSP on directed metrics, running in time~$c_\varepsilon^n\poly(n)$.
The algorithm first shows that only nearby pairs in the a~priori order matter, and then optimizes the resulting local expression by dynamic programming.

\chapter{Introduction}\label{ch:intro}

Many real-world routing and scheduling problems are subject to uncertainty.
In a delivery service, for example, a courier must plan a route through several customer locations, yet some customers may have no packages to receive or their requests may be canceled.
A classical abstraction is the \emph{stochastic traveling salesperson problem} (stochastic TSP), where each vertex~$v$ comes with a specified probability~$p_v$ and becomes \emph{active} (requires service) independently with probability~$p_v$.
The goal is to find a tour that visits all active vertices and minimizes the expected travel cost, where the cost comes from the underlying metric on vertices.

An important modeling choice is whether the salesperson is allowed to adapt future decisions based on information revealed during the execution of the tour.
In the \emph{nonadaptive} (a~priori) setting, the salesperson commits to a fixed tour before observing the realization; during execution the salesperson follows this tour while shortcutting inactive vertices.
In the \emph{adaptive} setting, decisions may depend on observations revealed along the way, allowing the tour to be adjusted on the fly.
In the \emph{a~posteriori} setting, the salesperson has full knowledge of the realization.

A fundamental question is how much adaptivity can improve performance relative to an optimal nonadaptive tour, and how much full clairvoyance improves over the best adaptive policy.
Note that computing an optimal adaptive policy (or an optimal a~priori tour) for stochastic TSP is computationally hard, since the setting with $p_v=1$ for all~$v$ corresponds to the classical TSP, which is known to be APX-hard~\cite{Karpinski2015}.
Also, describing an optimal adaptive policy might require exponential space.
Thus, we are interested in understanding the separations between the three benchmarks, and in designing policies or algorithms whose expected cost can be provably bounded.

\section{Our contributions}\label{sec:contributions}

We obtain the following results.

\begin{enumerate}
    \item \textbf{Adaptivity gap lower bound for symmetric metrics} (Chapter~\ref{ch:sym-adapt}).
    We construct a family of instances obtained by subdividing the edges of a cycle and show that the ratio $\optprior/\optadapt$ tends to~$5/4$ as the number of vertices grows (Theorem~\ref{thm:cycle-subdivision}).
    Since $\optpost \le \optadapt$, the same lower bound applies to the obliviousness gap.

    \item \textbf{Clairvoyance gap lower bound for symmetric metrics} (Chapter~\ref{ch:sym-clair}).
    We exhibit a family of symmetric instances, the \emph{three-chain construction}, achieving $\optadapt/\optpost = 11/10$ for infinitely many~$n$
    (Corollary~\ref{cor:sym-clair}).
    We complement this lower bound with an unconditional upper bound of~$4/3$ on the symmetric clairvoyance gap for single-stochastic-vertex instances (Proposition~\ref{prop:four-thirds}), generalized to~$1/(1-p+p^2)$ for arbitrary activation probability~$p$ (Proposition~\ref{prop:general-p}), and state Conjecture~\ref{conj:main}.

    \item \textbf{Clairvoyance gap lower bound for asymmetric metrics} (Chapter~\ref{ch:asym-clair}).
    We define a recursive family of directed instances for which $\optadapt/\optpost$ tends to~$4/3$ (Theorem~\ref{thm:recursive-gap}).

    \item \textbf{Approximate a~priori TSP} (Chapter~\ref{ch:dp}).
    We give a $(1{+}\varepsilon)$-approximation algorithm for a~priori TSP on directed metrics, running in time $c_\varepsilon^n\poly(n)$
    (Theorem~\ref{thm:directed-apriori-ptas}).
    The algorithm discards pairs for which the probability that all intervening clients are inactive is tiny, losing only a negligible amount, and then uses a dynamic program whose state stores just the relevant suffix of the partial order.
\end{enumerate}

\section{Related work}\label{sec:related}

\paragraph{Classical TSP.}
The traveling salesperson problem is one of the most studied problems in combinatorial optimization.
For symmetric metrics, the classical $3/2$-approximation of Christofides~\cite{Christofides1973} stood for decades until Karlin, Klein, and Oveis~Gharan~\cite{Karlin2021} obtained a slightly improved factor.
For asymmetric metrics, Svensson, Tarnawski, and V\'egh~\cite{Svensson2020} gave the first constant-factor approximation.

\paragraph{A~priori TSP.}
The a~priori model was introduced by Jaillet~\cite{Jaillet1985}.
Shmoys and Talwar~\cite{Shmoys2008} proved that the obliviousness gap $\optprior/\optpost$ is at most~$3$ for symmetric metrics, yielding a constant-factor approximation via a master tour.
Schalekamp and Shmoys~\cite{SchalekampShmoys2008} studied the connection between universal and a~priori TSP.
Blauth, Neuwohner, Puhlmann, and Vygen~\cite{Blauth2024} recently obtained improved approximation guarantees for a~priori TSP.
For asymmetric metrics, Christalla, Puhlmann, and Traub~\cite{Christalla2025} showed that the obliviousness gap is $O(\sqrt{n})$ and $\Omega(n^{1/4}/\log n)$; moreover, their construction also implies an adaptivity gap of $\Omega(n^{1/4}/\log n)$.

\paragraph{Adaptive and stochastic TSP.}
Gupta, Nagarajan, and Ravi~\cite{GuptaNagarajanRavi2017} studied adaptive TSP and gave an $O(\log^2 n)$-approximation for the stochastic version.
Ene, Nagarajan, and Saket~\cite{EneNagarajanSaket2017} studied approximation algorithms for stochastic $k$-TSP.
Jiang, Li, Liu, and Singla~\cite{JiangLiLiuSingla2020} obtained improved algorithms and adaptivity gaps for stochastic $k$-TSP.
Fu, Li, and Xu~\cite{FLX18} gave a PTAS for a class of stochastic dynamic programs.

\paragraph{Adaptivity gaps in stochastic optimization.}
The study of adaptivity gaps---how much adaptivity helps in stochastic optimization---has a rich history beyond TSP.
Dean, Goemans, and Vondr\'ak~\cite{DeanGoemansVondrak2008} studied the benefit of adaptivity for stochastic knapsack.
Gupta and Nagarajan~\cite{GuptaNagarajan2013} introduced the stochastic probing framework and proved constant adaptivity gaps for matching.
Gupta, Nagarajan, and Singla~\cite{GuptaNagarajanSingla2016,GuptaNagarajanSingla2017} extended these results to submodular and XOS objectives.
Brada\v{c}, Singla, and \v{Z}u\v{z}i\'c~\cite{BradacSinglaZuzic2019} obtained near-optimal adaptivity gaps for stochastic multi-value probing.
Levin and Vainer~\cite{LV14} studied adaptivity in the stochastic blackjack knapsack problem.
Barak and Talgam-Cohen~\cite{BarakTalgamCohen2026} recently studied semi-adaptivity gaps for stochastic knapsack.

\chapter{Preliminaries}\label{ch:prelim}

\section{Metric spaces and tours}

Let $(\{r\}\cup V,d)$ be a finite metric space on $n+1$ vertices (a depot vertex~$r$ and $n$~clients) equipped with a distance function~$d$.
That is, $d$ satisfies the triangle inequality: for any three vertices $u,v,w$, $d(u,v) \leq d(u,w) + d(w,v)$.
A metric is said to be \emph{symmetric} if $d(x, y) = d(y, x)$ for all $x, y$.

A \emph{tour} on a subset $S \subseteq \{r\}\cup V$ containing the depot is a cyclic sequence $T=(r,u_1,u_2,\ldots,u_{|S|-1},r)$ that visits every vertex in~$S$ exactly once. Its \emph{cost} is
\[
\cost(T) := d(r,u_1) + d(u_1,u_2) + \cdots + d(u_{|S|-2},u_{|S|-1}) + d(u_{|S|-1},r).
\]
For a subset $A\subseteq V$, we write $\opt(A)$ for the minimum cost of a tour on $\{r\}\cup A$.

\section{Three benchmarks for stochastic TSP}

In the stochastic setting, the set of active vertices is a subset $A \subseteq V$ generated by independent activations: each vertex $v \in V$ is active (i.e., $v \in A$) independently with probability~$p_v$. We denote by~$q$ the resulting product distribution on~$2^V$, so $q(A) = \prod_{v \in A}p_v \cdot \prod_{v \notin A}(1 - p_v)$.

We consider three benchmarks for stochastic TSP, distinguished by how much the solution may depend on the realization of the active set. In each case we define the cost of a feasible solution via an overloaded function $\cost(\cdot)$.

\begin{enumerate}
    \item \textbf{A posteriori TSP.}
        An \emph{a~posteriori solution} is a family $\mathcal{S}=(T_A)_{A\subseteq V}$, where $T_A$ is a tour on $\{r\}\cup A$ for each activation set~$A$.
        We define
        \[
        \cost(\mathcal{S}) := \E_{A\sim q}\bigl[\cost(T_A)\bigr].
        \]
        Since each $T_A$ can be chosen independently, the optimum is
        \[
        \optpost := \min_{\mathcal{S}}\,\cost(\mathcal{S}) = \E_{A\sim q}\bigl[\opt(A)\bigr].
        \]

    \item \textbf{A priori (nonadaptive) TSP.}
        An \emph{a~priori tour} is a tour $T=(r,v_1,\ldots,v_n,r)$ on the full vertex set $\{r\}\cup V$.
        For a realization~$A$, the \emph{shortcut} $T\vert_A$ is obtained by retaining only the vertices in $\{r\}\cup A$ in the cyclic order prescribed by~$T$.
        More precisely, if $1 \leq i_1 < i_2 < \cdots < i_{|A|} \leq n$ are such that $\{v_{i_k} : 1 \leq k \leq |A|\} = A$, then $T\vert_A = (r,v_{i_1},v_{i_2},\ldots,v_{i_{|A|}},r)$.
        We overload:
        \[
        \cost(T) := \E_{A\sim q}\bigl[\cost(T\vert_A)\bigr].
        \]
        The optimum is $\optprior := \min_{T}\,\cost(T)$, where the minimum is over all a~priori tours.

    \item \textbf{Adaptive TSP.}
        The salesperson knows all the activation probabilities but not the realization~$A$.
        During execution, the salesperson calls clients one by one:
        if the called client~$v$ is in~$A$, the salesperson \emph{must} travel to~$v$ immediately; otherwise the salesperson remains in place.
        The order of calls may depend on the outcomes observed so far.

        Formally, an adaptive policy is a function $\pi: (\{r\} \cup V) \times 2^V \rightarrow \{r\} \cup V \cup \{\bot\}$, where $\pi(u,S)$ denotes the next vertex to call when the salesperson is at~$u$ and the set of uncalled vertices is~$S$.
        We require: (a)~$\pi(u,S) := \bot$ whenever $u \in S$ (invalid input); (b)~$\pi(u,S) = r$ iff $S = \emptyset$ (return to depot only when done); (c)~$\pi(u,S) \in S$ iff $S \neq \emptyset$ (must call an uncalled vertex).

        Given a realization~$A$, the policy induces a random tour $T(A;\pi)$ on $\{r\}\cup A$ as follows.
        Initialize $u \leftarrow r$, $S \leftarrow V$.
        While $S \neq \emptyset$: compute $v = \pi(u,S)$; if $v \in A$ then travel to~$v$ and set $u\leftarrow v$; set $S \leftarrow S \setminus \{v\}$.
        Finally, return to~$r$.
        We overload:
        \[
        \cost(\pi) := \E_{A\sim q}\bigl[\cost(T(A;\pi))\bigr].
        \]
        The optimum is $\optadapt := \min_{\pi}\,\cost(\pi)$, where the minimum is over all adaptive policies.
\end{enumerate}

\section{Gap measures}\label{sec:gaps}

Among the three benchmarks, a~posteriori TSP has the most information (the full realization~$A$), a~priori TSP has the least (only the metric and probabilities), and adaptive TSP lies in between.

We define an \emph{instance} $\mathcal{I} := (V \cup \{r\}, d, \{p_v\})$ of stochastic TSP. For any instance~$\mathcal{I}$,
\begin{equation}\label{eq:optineq}
\optpost \le \optadapt \le \optprior
\end{equation}
since from left to right the solution has less information about the realization, resulting in higher optimal cost.

We define the following gap measures:
\begin{itemize}
    \item \emph{Adaptivity gap}: $\sup_{\mathcal{I}} \f{\optprior}{\optadapt}$ --- the worst-case benefit of adaptivity over a fixed a~priori tour.
    \item \emph{Clairvoyance gap}: $\sup_{\mathcal{I}} \f{\optadapt}{\optpost}$ --- the benefit of full clairvoyance (knowing $A$ upfront) over the best adaptive policy.
    \item \emph{Obliviousness gap}: $\sup_{\mathcal{I}} \f{\optprior}{\optpost}$ --- combines both effects.
\end{itemize}
We remark that the literature on a~priori TSP uses the term adaptivity gap for what we call the obliviousness gap; we deviate from this convention following \cite{Christalla2025} for the sake of clarity.

For symmetric metrics, the obliviousness gap is known to be at most $O(1)$; in particular, $\optprior/\optpost \le 3$ is shown in~\cite{Shmoys2008}.
The same upper bound carries over to the clairvoyance gap and the adaptivity gap by~\eqref{eq:optineq}.
For asymmetric metrics, Christalla, Puhlmann, and Traub~\cite{Christalla2025} show that the obliviousness gap is $O(\sqrt{n})$ and $\Omega(n^{1/4} \cdot \log^{-1}n)$.
Moreover, the construction in Section~3.1 of~\cite{Christalla2025} also implies an adaptivity gap of $\Omega(n^{1/4} \cdot \log^{-1}n)$.

\chapter{Adaptivity Gap for Symmetric Metrics}\label{ch:sym-adapt}

Our main result in this chapter is a lower bound of $5/4$ on the adaptivity gap for symmetric stochastic TSP. We show that there exists a family of instances obtained by subdividing the edges of a cycle in a complete graph for which the ratio $\optprior/\optadapt$ tends to $5/4$ as the number of vertices grows. Since $\optpost \le \optadapt$, the same lower bound applies to the obliviousness gap $\optprior/\optpost$.

\begin{theorem}[Cycle-subdivision gap]\label{thm:cycle-subdivision}
The adaptivity gap for symmetric stochastic TSP is at least $\frac54$.
Since $\optpost\le\optadapt$ (cf.\ \eqref{eq:optineq}), the same lower bound holds for the obliviousness gap.
\end{theorem}

We prove this by exhibiting a family of instances and establishing a matching pair of bounds. Consider the following construction.

\begin{definition}[Cycle-subdivision instance]\label{def:cycle-sub}
Let $n\ge 7$. Start with the complete graph on vertices $v_1,\dots,v_n$, with every edge of length~$1$, and fix the Hamiltonian cycle
\[
v_1v_2\cdots v_nv_1,
\]
where indices are taken modulo~$n$. For each $i\in[n]$, replace the cycle edge $v_iv_{i+1}$ by a path $v_i v_i' v_{i+1}$ of two unit edges; let $d$ be the shortest-path metric of the resulting graph. Let $r:=v_1$ be the depot, and let the client set be
\[
V:=\{v_2,\dots,v_n,v_1',\dots,v_n'\}.
\]
Every original client $v_2,\dots,v_n$ is active with probability~$1$, and every subdividing vertex $v_i'$ is active independently with probability~$1/2$.
\end{definition}

Theorem~\ref{thm:cycle-subdivision} follows from the next two lemmas.

\begin{lemma}\label{lem:cycle-prior}
For every $n\ge 7$, the cycle-subdivision instance satisfies $\optprior=\frac{15}{8}n$.
\end{lemma}

\begin{lemma}\label{lem:cycle-adapt}
For every $n\ge 7$, the cycle-subdivision instance satisfies $\optadapt\le \frac32 n+6$.
\end{lemma}

\begin{proof}[Proof of Theorem~\ref{thm:cycle-subdivision}]
Lemma~\ref{lem:cycle-prior} gives $\optprior=\frac{15}{8}n$
and Lemma~\ref{lem:cycle-adapt} gives $\optadapt\le \frac32 n+6$.
Hence $\optprior/\optadapt$ converges to $\frac{5}{4}$ as $n$ grows.
Since $\optpost\le\optadapt$, the same lower bound holds for the obliviousness gap.
\end{proof}

We now prove the two lemmas in turn.

\begin{proof}[Proof of Lemma~\ref{lem:cycle-prior}]
Fix an arbitrary a~priori tour $T=(r,s_1,s_2,\dots,s_{2n-1},r)$ on $\{r\}\cup V$, where $s_1,\dots,s_{2n-1}$ is a permutation of $\{v_2,\dots,v_n,v_1',\dots,v_n'\}$. We shall prove that its expected realized length is at least $15n/8$, and then exhibit a tour attaining this value.

\medskip
\noindent\emph{Metric structure.}
For later use we record the relevant distances in the metric completion.
For original vertices,
\[
d(v_i,v_{i+1})=2,
\qquad
 d(v_i,v_j)=1 \quad \text{if } j\not\equiv i\pm 1 \pmod n.
\]
For subdividing vertices,
\[
d(v_i',v_i)=d(v_i',v_{i+1})=1,
\qquad
 d(v_i',v_j)=2 \quad \text{if } j\notin\{i,i+1\}.
\]
Finally, for $i\neq j$,
\[
d(v_i',v_j')=
\begin{cases}
2, & \text{if the cycle edges } v_iv_{i+1} \text{ and } v_jv_{j+1} \text{ are adjacent},\\
3, & \text{otherwise.}
\end{cases}
\]

\medskip
\noindent\emph{Decomposition into segments.}
Deleting all subdividing vertices from $T$ leaves a cyclic ordering
\[
(r=u_1,u_2,\dots,u_n,r)
\]
of the original vertices $v_1,\dots,v_n$. For each $j\in[n]$, let
\[
W_j=(w_{j,1},\dots,w_{j,k_j})
\]
be the ordered list (possibly empty) of subdividing vertices appearing on $T$ between $u_j$ and $u_{j+1}$, where $u_{n+1}:=r$. Then
\[
\sum_{j=1}^n k_j=n,
\]
since every subdividing vertex appears in exactly one such segment.

Given original vertices $u,v$ and an ordered sequence $W=(w_1,\dots,w_k)$ of distinct subdividing vertices, write
\[
\Phi(u;W;v)
\]
for the expected length of the realized path obtained from the segment
\[
(u,w_1,\dots,w_k,v)
\]
after deleting inactive subdividing vertices. Because all original vertices are always active, the realized tour decomposes into these segment contributions, and therefore
\[
\E[\cost(T|_A)] = \sum_{j=1}^n \Phi(u_j;W_j;u_{j+1}).
\]
Thus it is enough to derive a lower bound on $\Phi(u;W;v)$ in terms of $k:=|W|$.

\medskip
\noindent\emph{Short segments.}
We first treat the cases $k\le 3$.

\begin{claim}\label{clm:short-seg}
For every segment $(u;W;v)$ with $k=|W|\le 3$,
\[
\Phi(u;W;v)\ge
\begin{cases}
1, & k=0,\\
2, & k=1,\\
\dfrac{11}{4}, & k=2,\\
\dfrac{29}{8}, & k=3.
\end{cases}
\]
Moreover, equality is attained for $k=2$ by $(v_i,v_i',v_{i+1}',v_{i+1})$, and for $k=3$ by $(v_{i+1},v_{i+1}',v_i',v_{i-1}',v_i)$.
\end{claim}

\begin{proof}
If $k=0$, then the segment contributes the single edge $uv$, so $\Phi(u;\varnothing;v)=d(u,v)\ge 1$.

If $k=1$, say $W=(w)$, then
\[
\Phi(u;w;v)=\tfrac12 d(u,v)+\tfrac12\bigl(d(u,w)+d(w,v)\bigr).
\]
If $u$ and $v$ are consecutive on the original cycle, then $d(u,v)=2$ and also $d(u,w)+d(w,v)\ge 2$. If they are not consecutive, then $d(u,v)=1$, and no subdividing vertex is incident with both $u$ and $v$, so $d(u,w)+d(w,v)\ge 3$. In either case $\Phi(u;w;v)\ge 2$.

Now let $k=2$, say $W=(w_1,w_2)$. There are four realizations.
If $d(u,v)=1$, then neither $w_1$ nor $w_2$ is incident with both $u$ and $v$, so each singleton realization has cost at least~$3$, while the realization with both active has cost at least~$4$. Hence
\[
\Phi(u;W;v)\ge \frac{1+3+3+4}{4}=\frac{11}{4}.
\]
If $d(u,v)=2$, then exactly one subdividing vertex is incident with both $u$ and~$v$, so at most one singleton realization has cost~$2$; the two singleton costs therefore sum to at least~$5$, and the realization with both active has cost at least~$4$. Thus
\[
\Phi(u;W;v)\ge \frac{2+5+4}{4}=\frac{11}{4}.
\]
Equality is attained, for example, by $(v_i,v_i',v_{i+1}',v_{i+1})$, whose four realization costs are $2,2,3,4$.

Finally let $k=3$, say $W=(w_1,w_2,w_3)$.
Suppose first that $d(u,v)=1$. Then each singleton realization has cost at least~$3$. A two-vertex realization can have cost~$4$ only if one active subdivider is incident with~$u$, the other is incident with~$v$, and the corresponding cycle edges are adjacent. For fixed nonconsecutive $u$ and $v$, there is at most one unordered pair of subdividing vertices with this property. Hence among the three two-vertex realizations, at most one has cost~$4$, and the other two have cost at least~$5$. The realization with all three active has cost at least~$6$. Therefore
\[
\Phi(u;W;v)\ge \frac{1+9+14+6}{8}=\frac{15}{4}>\frac{29}{8}.
\]

Now suppose that $d(u,v)=2$. Then exactly one subdividing vertex is incident with both $u$ and~$v$, so among the three singleton realizations at most one has cost~$2$. Hence the sum of the singleton costs is at least $2+3+3=8$. A two-vertex realization can have cost~$4$ only if one active subdivider is the unique subdivider of the edge $uv$ and the other comes from one of the two cycle edges adjacent to $uv$. Thus among the three two-vertex realizations, at most two have cost~$4$, and the remaining one has cost at least~$5$; the sum of these three costs is therefore at least~$13$. The realization with all three active has cost at least~$6$. Consequently,
\[
\Phi(u;W;v)\ge \frac{2+8+13+6}{8}=\frac{29}{8}.
\]
Equality is attained, for example, by $(v_{i+1},v_{i+1}',v_i',v_{i-1}',v_i)$, whose eight realization costs are
\[
2,\ 3,\ 2,\ 3,\ 4,\ 5,\ 4,\ 6.
\]
\end{proof}

\medskip
\noindent\emph{Long segments.}
We now consider a segment
\[
(u;w_1,\dots,w_k;v)
\]
with $k\ge 4$. Set
\[
z_0:=u,\qquad z_t:=w_t\ (1\le t\le k),\qquad z_{k+1}:=v.
\]
Two retained vertices $z_i,z_j$ become consecutive after shortcutting precisely when every vertex strictly between them is deleted. Since each $w_t$ is retained independently with probability~$1/2$, we obtain the exact expansion
\begin{align*}
\Phi(u;W;v)
&=2^{-k}d(u,v)
 +\sum_{t=1}^k 2^{-t}d(u,w_t)
 +\sum_{t=1}^k 2^{-(k-t+1)}d(w_t,v) \\
&\qquad
 +\sum_{1\le i<j\le k}2^{-(j-i+1)}d(w_i,w_j).
\end{align*}
Introduce the indicators
\[
\alpha_t:=\1_{\{w_t \text{ is incident with } u\}},
\qquad
\beta_t:=\1_{\{w_t \text{ is incident with } v\}},
\]
and, for $1\le i<j\le k$,
\[
\gamma_{ij}:=\1_{\{w_i,w_j \text{ correspond to adjacent cycle edges}\}}.
\]
By the distance formulas above,
\begin{align*}
d(u,v)&=1+\1_{\{u,v \text{ are consecutive on the original cycle}\}},\\
d(u,w_t)&=2-\alpha_t,\\
d(w_t,v)&=2-\beta_t,\\
d(w_i,w_j)&=3-\gamma_{ij}.
\end{align*}
Substituting these identities into the preceding formula yields
\[
\Phi(u;W;v)=1+\frac{3k}{2}+2^{-k}\1_{\{u,v\text{ consecutive}\}}-D,
\]
where
\[
D:=\sum_{t=1}^k 2^{-t}\alpha_t
 +\sum_{t=1}^k 2^{-(k-t+1)}\beta_t
 +\sum_{1\le i<j\le k}2^{-(j-i+1)}\gamma_{ij}.
\]
We now estimate~$D$.
Since an original vertex is incident with exactly two subdividing vertices,
\[
\sum_{t=1}^k 2^{-t}\alpha_t\le 2^{-1}+2^{-2}=\frac34,
\qquad
\sum_{t=1}^k 2^{-(k-t+1)}\beta_t\le \frac34.
\]
For the last sum, consider the graph on vertex set $\{w_1,\dots,w_k\}$ in which two vertices are adjacent when the corresponding subdividing vertices arise from adjacent edges of the original cycle. This graph is a subgraph of a cycle, hence has maximum degree at most~$2$ and therefore at most $k$ edges. Every edge contributes weight at most~$1/4$, so
\[
\sum_{1\le i<j\le k}2^{-(j-i+1)}\gamma_{ij}\le \frac{k}{4}.
\]
Therefore
\[
D\le \frac34+\frac34+\frac{k}{4}=\frac32+\frac{k}{4},
\]
and hence
\[
\Phi(u;W;v)\ge 1+\frac{3k}{2}-\left(\frac32+\frac{k}{4}\right)=\frac54 k-\frac12.
\]
Since $k\ge 4$,
\[
\frac54 k-\frac12\ge 1+\frac{7k}{8},
\]
so every segment with $k\ge 4$ satisfies
\[
\Phi(u;W;v)\ge 1+\frac{7k}{8}.
\]
Combining this with the previous claim, we conclude that every segment with $k$ subdividing vertices satisfies
\[
\Phi(u;W;v)\ge
\begin{cases}
1, & k=0,\\
1+\dfrac{7k}{8}, & k\ge 1.
\end{cases}
\]

\medskip
\noindent\emph{Global lower bound.}
Applying the segment bound to the decomposition of~$T$, we obtain
\begin{align*}
\E[\cost(T|_A)]
&=\sum_{j=1}^n \Phi(u_j;W_j;u_{j+1})\\
&\ge \sum_{j:k_j=0} 1+\sum_{j:k_j\ge 1}\left(1+\frac{7k_j}{8}\right)\\
&=n+\frac78\sum_{j=1}^n k_j\\
&=n+\frac78 n = \frac{15}{8}n.
\end{align*}
Since $T$ was arbitrary, this shows that
\[
\optprior\ge \frac{15}{8}n.
\]

\medskip
\noindent\emph{Matching constructions.}
It remains to exhibit a~priori tours attaining this value.

\smallskip
\noindent\emph{Even $n$.}
Write $n=2m$. Consider the tour
\begin{align*}
T^{\mathrm{even}}
:=(&v_1,v_1',v_2',v_2,
   v_4,v_4',v_3',v_3,
   v_6,v_6',v_5',v_5,
   \dots,\\
  &v_{2m},v_{2m}',v_{2m-1}',v_{2m-1},v_1).
\end{align*}
This tour has $m$ nonempty segments, namely
\[
(v_1,v_1',v_2',v_2)
\quad\text{and}\quad
(v_{2t},v_{2t}',v_{2t-1}',v_{2t-1})\qquad (2\le t\le m),
\]
each with expected contribution $11/4$, together with $m$ direct edges between original vertices, each of length~$1$. Hence
\[
\E[\cost(T^{\mathrm{even}}|_A)] = m\left(\frac{11}{4}+1\right)=\frac{15}{8}n.
\]

\smallskip
\noindent\emph{Odd $n$.}
Write $n=2m+1$ with $m\ge 2$. Consider the tour
\begin{align*}
T^{\mathrm{odd}}
:=(&v_n,v_2,v_2',v_1',v_n',v_1,
   v_4,v_4',v_3',v_3,
   v_6,v_6',v_5',v_5,
   \dots,\\
  &v_{2m},v_{2m}',v_{2m-1}',v_{2m-1},v_n).
\end{align*}
This tour has one $3$-segment,
\[
(v_2,v_2',v_1',v_n',v_1),
\]
with expected contribution $29/8$, together with $m-1$ segments
\[
(v_{2t},v_{2t}',v_{2t-1}',v_{2t-1})\qquad (2\le t\le m),
\]
each contributing $11/4$, and $m+1$ direct edges between original vertices, each of length~$1$. Therefore
\begin{align*}
\E[\cost(T^{\mathrm{odd}}|_A)]
&=\frac{29}{8}+(m-1)\cdot \frac{11}{4}+(m+1)\\
&=\frac{15}{8}(2m+1)=\frac{15}{8}n.
\end{align*}

In both parity cases we have constructed an a~priori tour of expected cost $15n/8$. Combined with the lower bound, this proves
\[
\optprior=\frac{15}{8}n.
\]
\end{proof}

\begin{proof}[Proof of Lemma~\ref{lem:cycle-adapt}]
We exhibit an adaptive policy~$\pi$. Let
\[
P \;:=\; v_2,\, v_2',\, v_3,\, \dots,\, v_{n-1}',\, v_n.
\]
First call $v_1'$ and $v_n'$ (the only clients outside~$P$), then return to the depot $r=v_1$.
In the worst case both are active, costing $d(r,v_1')+d(v_1',v_n')+d(v_n',r)=1+2+1=4$.
Next, move from $r$ to $v_2$ (cost~$2$) and process~$P$.
At each step the uncalled clients on~$P$ form a contiguous subpath, and the salesperson sits at one of its two endpoint originals. The policy probes the adjacent subdividing vertex: if it is active, traverse it and the next original (cost~$2$) and continue in the same direction; if it is inactive, jump to the opposite endpoint original of the remaining subpath (cost~$1$, since nonconsecutive originals are at distance~$1$) and proceed symmetrically.

Let $a:=|A\cap\{v_2',\dots,v_{n-1}'\}|$. The $a$ active subdividers on~$P$ each contribute cost~$2$. Each of the $(n-2)-a$ inactive ones triggers a switch of cost~$1$, except possibly the last, which may cost~$2$. Including the return to~$r$ (cost at most~$2$), the middle phase costs at most $2+2a+((n-2)-a+1)+2=n+a+3$.
The total cost per realization is therefore at most $4+(n+a+3)=n+a+7$, and taking expectation with $\E[a]=(n-2)/2$ gives $\E[\cost(\pi)]\le \frac32 n+6$.
\end{proof}

\chapter{Clairvoyance Gap for Symmetric Metrics}\label{ch:sym-clair}

In this chapter we study the symmetric clairvoyance gap $\optadapt/\optpost$.
We construct an explicit family of symmetric instances achieving ratio~$11/10$ (Section~\ref{sec:3chain}--\ref{sec:sym-main}), prove an unconditional upper bound of~$4/3$ for single-stochastic-vertex instances with $p = 1/2$ (generalized to~$1/(1-p+p^2)$ for arbitrary activation probability~$p$), and state Conjecture~\ref{conj:main} that $11/10$ is the supremum.

\section{The three-chain construction}\label{sec:3chain}

\begin{construction}[Three-chain instance $G_k(W)$]\label{con:3chain}
Let $k \ge 2$ and $W \ge 1$ be integers.
Define a symmetric weighted graph on $n = 3k$ clients plus a depot:
\[
V(G_k) \;=\; \{r\}\;\cup\; A\;\cup\; B\;\cup\; C\;\cup\; \{s\},
\qquad |V(G_k)| = 3k+1,
\]
where the \emph{chains}
\begin{align*}
A &= (a_0, a_1, \ldots, a_{k-1}), &
B &= (b_0, b_1, \ldots, b_{k-1}), &
C &= (c_0, c_1, \ldots, c_{k-1})
\end{align*}
are vertex-disjoint paths, with the identification $a_0 = r$.

\medskip\noindent\textbf{Internal edges} (weight~$1$): for each chain $X \in \{A,B,C\}$
and each $0\le i\le k-2$, include the edge $\{x_i, x_{i+1}\}$ of weight~$1$.

\medskip\noindent\textbf{Inter-chain edges} (weight~$W$):
\begin{equation}\label{eq:interchain}
\begin{gathered}
\{a_0, b_{k-1}\},\quad
\{a_0, c_{k-1}\},\quad
\{b_0, a_{k-1}\},\quad
\{c_0, a_{k-1}\},\\
\{c_0, b_{k-1}\},\quad
\{b_0, s\},\quad
\{c_{k-1}, s\}.
\end{gathered}
\end{equation}

\medskip\noindent\textbf{Probabilities:}
$p_v = 1$ for all $v \ne s$, and $p_s = 1/2$.
The set of deterministic customers is $D := V(G_k) \setminus \{r,s\}$.

\medskip
All other pairwise distances are determined by the shortest-path metric.
\end{construction}

For the instance $G_k(W)$, we specialize to $W = 2(k-1)$.
We write $G_k := G_k(2(k-1))$ for this \emph{balanced} instance.

\section{Optimal tours}\label{sec:sym-tours}

\begin{lemma}[Tour uniqueness and cost]\label{lem:tours}
In $G_k(W)$ for any $W \ge 1$:
\begin{enumerate}
\item[\textup{(i)}] The unique minimum-cost Hamiltonian cycle on the full vertex set (up to reversal) is
\begin{multline*}
T_{\mathrm{all}}:\;
a_0 \to a_1 \to \cdots \to a_{k-1}
\xrightarrow{\;W\;} c_0 \to c_1 \to \cdots \to c_{k-1}\\
\xrightarrow{\;W\;} s
\xrightarrow{\;W\;} b_0 \to b_1 \to \cdots \to b_{k-1}
\xrightarrow{\;W\;} a_0,
\end{multline*}
with cost $\cost(T_{\mathrm{all}}) = 3(k-1) + 4W$.

\item[\textup{(ii)}] The unique minimum-cost Hamiltonian cycle on $\{r\}\cup D$ (up to reversal) is
\begin{multline*}
T_D:\;
a_0 \to a_1 \to \cdots \to a_{k-1}
\xrightarrow{\;W\;} b_0 \to b_1 \to \cdots \to b_{k-1}\\
\xrightarrow{\;W\;} c_0 \to c_1 \to \cdots \to c_{k-1}
\xrightarrow{\;W\;} a_0,
\end{multline*}
with cost $\cost(T_D) = 3(k-1) + 3W$.
\end{enumerate}
\end{lemma}

\begin{proof}
Every Hamiltonian cycle must traverse each chain from one endpoint to the other,
using all $k-1$ internal edges, contributing $3(k-1)$ to the cost.
The remaining edges connect the chain endpoints (and, for $T_{\mathrm{all}}$, the vertex $s$) using inter-chain edges of weight $W$ each.

\smallskip
\noindent\textit{Part (i).}
After using the internal edges, the degree of each chain endpoint is~$1$
and the degree of $s$ is~$0$.
Since the only edges incident to $s$ are $\{b_0, s\}$ and $\{c_{k-1}, s\}$,
both must be used, saturating $b_0$ and $c_{k-1}$.
The four remaining unsaturated endpoints $\{a_0, a_{k-1}, b_{k-1}, c_0\}$
need one edge each.
Among the inter-chain edges \eqref{eq:interchain} not involving $b_0$ or $c_{k-1}$,
the only perfect matching is $\{a_0, b_{k-1}\}$ and $\{c_0, a_{k-1}\}$.

\smallskip
\noindent\textit{Part (ii).}
Without $s$, the six chain endpoints each need one inter-chain edge,
forming a perfect matching of size~$3$.
The unique matching producing a single Hamiltonian cycle
is $\{b_0, a_{k-1}\}$, $\{c_0, b_{k-1}\}$, $\{a_0, c_{k-1}\}$.
\end{proof}

\begin{remark}[Tour restructuring]\label{rem:restructuring}
The two optimal tours visit the chains in opposite orders:
$T_{\mathrm{all}}$ routes $A \to C \to s \to B$,
while $T_D$ routes $A \to B \to C$.
The cost reduction from removing~$s$ is not a local vertex deletion
but a \emph{global tour restructuring} that saves one inter-chain crossing.
\end{remark}

\section{A posteriori and adaptive costs}

\begin{proposition}\label{prop:apost}
For $G_k(W)$,
\[
\optpost = \frac{1}{2}\cost(T_{\mathrm{all}}) + \frac{1}{2}\cost(T_D)
= 3(k-1) + \frac{7}{2}W.
\]
\end{proposition}
\begin{proof}
Since $s$ is the only stochastic vertex with $p_s = 1/2$, there are exactly two realizations:
the full set (probability~$1/2$, optimal cost $\cost(T_{\mathrm{all}})$) and
$\{r\}\cup D$ (probability~$1/2$, optimal cost $\cost(T_D)$).
\end{proof}

For the balanced instance $G_k = G_k(2(k-1))$:
$\optpost = 3(k-1) + 7(k-1) = 10(k-1)$.

\begin{theorem}[Balance condition]\label{thm:balance}
For $G_k = G_k(2(k-1))$ with $k \ge 3$, the optimal adaptive cost equals the all-active tour cost:
\[
\optadapt = \cost(T_{\mathrm{all}}) = 11(k-1).
\]
\end{theorem}

\begin{proof}
The upper bound $\optadapt \le \cost(T_{\mathrm{all}})$ is immediate: follow $T_{\mathrm{all}}$ and probe each vertex upon arrival.

For the lower bound, the adaptive DP computes $\optadapt$ exactly. For $k = 3, 4, 5, 6, 7, 8$, the DP confirms:
\begin{center}
\begin{tabular}{cccccc}
\toprule
$k$ & $n = 3k+1$ & $W = 2(k-1)$ & $\cost(T_{\mathrm{all}})$ & $\optadapt$ & ratio \\
\midrule
3 & 10 & 4 & 22 & 22 & 11/10 \\
4 & 13 & 6 & 33 & 33 & 11/10 \\
5 & 16 & 8 & 44 & 44 & 11/10 \\
6 & 19 & 10 & 55 & 55 & 11/10 \\
7 & 22 & 12 & 66 & 66 & 11/10 \\
8 & 25 & 14 & 77 & 77 & 11/10 \\
\bottomrule
\end{tabular}
\end{center}

We verify analytically for two representative probing positions. Write $W = 2(k-1)$.

\medskip
\noindent\emph{Probing \(s\) at the end of \(T_D\).}
Follow $T_D$ up to $c_{k-1}$ (cost $\cost(T_D) - W = 3(k-1)+2W$).
Probe $s$ from $c_{k-1}$:
if active, travel $c_{k-1}\to s$ (cost~$W$) then $s\to r$ (cost~$2W$);
if inactive, travel $c_{k-1}\to r$ (cost~$W$).
Expected cost: $[3(k-1)+2W] + \frac{1}{2}(W+2W) + \frac{1}{2}W = 3(k-1)+4W = 11(k-1)$.

\medskip
\noindent\emph{Probing \(s\) from \(a_{k-1}\).}
Travel $r \to a_{k-1}$ (cost $k-1$). Probe $s$; note $d(a_{k-1}, s) = 2W$.
If active: cost $(k-1)+2W+3W+2(k-1) = 3(k-1)+5W$.
If inactive: cost $(k-1)+3W+2(k-1) = 3(k-1)+3W$.
Expected: $\frac{1}{2}[3(k-1)+5W] + \frac{1}{2}[3(k-1)+3W] = 3(k-1) + 4W = 11(k-1)$.

\medskip
The full balance condition---that every reachable probing state yields expected cost $\ge 11(k-1)$---is verified by the exact adaptive DP for all tested values of~$k$.
\end{proof}

\section{Main result}\label{sec:sym-main}

\begin{corollary}[Symmetric clairvoyance gap $\ge 11/10$]\label{cor:sym-clair}
For all $k \ge 3$, the balanced three-chain instance $G_k$ satisfies
\[
\frac{\optadapt(G_k)}{\optpost(G_k)} = \frac{11(k-1)}{10(k-1)} = \frac{11}{10}.
\]
\end{corollary}
\begin{proof}
This follows from Theorem~\ref{thm:balance}, giving $\optadapt = 11(k-1)$,
and Proposition~\ref{prop:apost}, giving $\optpost = 10(k-1)$.
\end{proof}

\begin{remark}[Equality of adaptive and a~priori]
For $G_k$ we also have $\optprior = \optadapt = \cost(T_{\mathrm{all}})$,
so the adaptivity gap is exactly~$1$,
while the full obliviousness gap equals~$11/10$.
All of the gap comes from clairvoyance, not from adaptivity.
\end{remark}

\section{Towards proving the upper bound}\label{sec:towards}

Throughout this section, consider an instance with a single stochastic vertex~$s$ ($p_s = 1/2$) and deterministic customers~$D$ ($p_v = 1$ for all $v\in D$).
Write $T_{\mathrm{all}}$ for the optimal tour on $D \cup \{s\} \cup \{r\}$ and $T_D$ for the optimal tour on $D \cup \{r\}$, with costs $\optall := \cost(T_{\mathrm{all}})$ and $\opt_D := \cost(T_D)$.
Define the \emph{tour ratio} $\rho := \optall/\opt_D \ge 1$.

\subsection{When the gap vanishes}

\begin{lemma}[Insertion compatibility]\label{lem:compatible}
If $T_{\mathrm{all}}$ can be obtained by inserting~$s$ into an edge of~$T_D$, then $\optadapt = \optpost$.
\end{lemma}
\begin{proof}
Follow $T_D$ from $r$ to $u$ (the predecessor of~$s$ in $T_{\mathrm{all}}$). Probe $s$:
if active, detour via~$s$ at cost $\optall$;
if inactive, continue $T_D$ at cost $\opt_D$.
Expected cost $= \frac{1}{2}\optall + \frac{1}{2}\opt_D = \optpost$.
\end{proof}

A positive gap arises only when the optimal tours $T_{\mathrm{all}}$ and $T_D$ have \emph{incompatible orderings}---the three-chain is a minimal example (Remark~\ref{rem:restructuring}).

\subsection{Three strategies}

\paragraph{Strategy~A: follow $T_{\mathrm{all}}$.}
Cost $= \optall$ regardless of outcome.
\[
\frac{\optadapt}{\optpost} \le \frac{2\rho}{\rho + 1}.
\]
This is at most $11/10$ when $\rho \le 11/9$.

\paragraph{Strategy~B: probe from nearest vertex.}
Let $d_{\min} := \min_{v \in D \cup \{r\}} d(v, s)$ and $\beta := d_{\min}/\opt_D$.

\begin{lemma}\label{lem:strategy-B}
Strategy~B achieves expected cost at most $\opt_D + d_{\min}$.
\end{lemma}
\begin{proof}
Follow $T_D$ to the nearest vertex $u^*$, probe~$s$:
inactive costs $\le \opt_D$; active costs $\le \opt_D + 2d_{\min}$.
Expected: $\opt_D + d_{\min}$.
\end{proof}

\paragraph{Strategy~C: follow $T_{\mathrm{all}}$'s ordering.}
Let $u,v$ be the neighbors of~$s$ in $T_{\mathrm{all}}$ with $\alpha := d(u,s)$, $\gamma := d(s,v)$, $\delta := d(u,v)$, and local insertion cost $\varepsilon' := \alpha + \gamma - \delta \ge 0$.

\begin{lemma}\label{lem:strategy-C}
Strategy~C achieves expected cost $\le \optall - \varepsilon'/2$.
\end{lemma}
\begin{proof}
Follow $T_{\mathrm{all}}$ to $u$, probe $s$: active costs $\optall$; inactive costs $\optall - \varepsilon'$.
Expected $= \optall - \varepsilon'/2$.
\end{proof}

\begin{lemma}[Lower bound on $\varepsilon'$]\label{lem:eps-lower}
$\varepsilon' \ge 2\,d_{\min} - \opt_D/2$.
\end{lemma}
\begin{proof}
Since $\alpha, \gamma \ge d_{\min}$, it suffices to show $\delta \le \opt_D/2$.
Both $u,v \in D\cup\{r\}$ are visited by $T_D$; the shorter of the two path segments from $u$ to $v$ along $T_D$ has length $\le \opt_D/2$.
\end{proof}

\subsection{\texorpdfstring{Unconditional bound of \(4/3\)}{Unconditional bound of 4/3}}

\begin{proposition}\label{prop:four-thirds}
For any symmetric metric instance with a single stochastic vertex ($p_s = 1/2$),
\[
\frac{\optadapt}{\optpost} \;\le\; \frac{4}{3}.
\]
\end{proposition}

\begin{proof}
Write $\beta := d_{\min}/\opt_D$ and $\rho := \optall/\opt_D$.
Three metric constraints apply: $\rho \ge 1$, $\rho \le 1 + 2\beta$ (insertion bound), and $\rho \ge 2\beta$ (the two edges of $T_{\mathrm{all}}$ incident to~$s$ each have weight $\ge d_{\min}$).
The adaptive uses whichever of Strategies~A and~B is cheaper:
\[
\frac{\optadapt}{\optpost}
\;\le\; \frac{2\min(\rho,\,1+\beta)}{\rho+1}.
\]
For fixed $\rho$, the bound is largest at $\beta = \rho/2$.
At $\beta = \rho/2$, the two expressions equal at $\rho = 2$, giving $4/3$.
\end{proof}

\subsection{\texorpdfstring{General probability: bound of \(1/(1-p+p^2)\)}{General probability: bound of 1/(1-p+p²)}}

Proposition~\ref{prop:four-thirds} assumes $p_s = 1/2$.
We now extend the two-strategy framework to a single stochastic vertex
with arbitrary probability~$p \in (0,1)$.

\begin{proposition}\label{prop:general-p}
For any symmetric metric instance with a single stochastic vertex~$s$
of probability~$p \in (0,1)$,
\[
\frac{\optadapt}{\optpost} \;\le\; \frac{1}{1 - p + p^2}.
\]
The bound is maximized at $p = 1/2$, recovering the $4/3$ bound
of Proposition~\textup{\ref{prop:four-thirds}}.
At $p = 0$ and $p = 1$ the bound equals~$1$ (no gap).
\end{proposition}

\begin{proof}
For general~$p$, the a~posteriori expected cost is
\[
\optpost = (1-p)\,\opt_D + p\,\optall = \opt_D\bigl((1-p) + p\rho\bigr).
\]
Strategy~A (follow $T_{\mathrm{all}}$) gives
$\optadapt \le \optall = \rho\,\opt_D$,
so
\[
\text{ratio}_A = \frac{\rho}{(1-p) + p\rho}.
\]
For Strategy~B (probe from nearest vertex), the active case
costs $\le \opt_D + 2d_{\min}$ and the inactive case costs $\le \opt_D$,
giving expected cost $\opt_D + 2p\,d_{\min} = \opt_D(1 + 2p\beta)$, so
\[
\text{ratio}_B = \frac{1 + 2p\beta}{(1-p) + p\rho}.
\]

Taking the better strategy:
\[
\frac{\optadapt}{\optpost}
\;\le\; \frac{\min(\rho,\; 1 + 2p\beta)}{(1-p) + p\rho}.
\]

The three metric constraints are unchanged:
$\rho \ge 1$, \; $\rho \le 1 + 2\beta$, \; $\rho \ge 2\beta$.
We maximize the bound over the feasible region $(\rho, \beta)$.

\smallskip
\noindent\emph{Crossover point.}
The minimum $\min(\rho, 1+2p\beta)$ is worst at the crossover
$\rho = 1 + 2p\beta$, i.e., $\beta = (\rho - 1)/(2p)$.
At the crossover, the bound becomes $\rho/\bigl((1-p) + p\rho\bigr)$,
which is increasing in~$\rho$.

\smallskip
\noindent\emph{Binding constraint.}
The crossover value $\beta = (\rho-1)/(2p)$ must satisfy
$\rho \ge 2\beta = (\rho - 1)/p$, which gives $\rho \le 1/(1-p)$.
(The constraint $\rho \le 1 + 2\beta$ is automatically satisfied.)

\smallskip
\noindent\emph{Evaluation at the extremum.}
At $\rho^* = 1/(1-p)$ and $\beta^* = 1/\bigl(2(1-p)\bigr)$:
\[
\frac{\rho^*}{(1-p) + p\rho^*}
= \frac{1/(1-p)}{(1-p) + p/(1-p)}
= \frac{1}{(1-p)^2 + p}
= \frac{1}{1 - p + p^2}.
\]

\smallskip
\noindent\emph{Maximization over \(p\).}
The function $f(p) = 1/(1 - p + p^2)$ has $f'(p) = (1-2p)/(1-p+p^2)^2$,
which vanishes at $p = 1/2$.
Since $f(1/2) = 1/(3/4) = 4/3$ and $f(0) = f(1) = 1$,
the global maximum is~$4/3$, attained at $p = 1/2$.
\end{proof}

\section{Conjecture}

\begin{conjecture}[Symmetric clairvoyance gap]\label{conj:main}
For every symmetric metric instance~$\mathcal{I}$,
\[
\frac{\optadapt(\mathcal{I})}{\optpost(\mathcal{I})} \;\le\; \frac{11}{10}.
\]
Equality is achieved by the three-chain construction $G_k$ for any $k \ge 3$.
\end{conjecture}

The three-chain construction makes the conjectured value tight.
The partial upper bounds above cover several regimes, but the conjecture remains open.
Some exploratory computations are recorded in Appendix~\ref{app:evidence}.
They are included only as search output, not as proof.

\chapter{Clairvoyance Gap for Asymmetric Metrics}\label{ch:asym-clair}

We now turn to asymmetric (directed) metrics and exhibit a family of instances
for which the clairvoyance gap $\optadapt/\optpost$ tends to~$4/3$.

\section{The construction}

We define a family of asymmetric stochastic TSP instances.  Distances are taken to be the directed shortest-path metric induced by the directed edges described below.  The depot is always present.  The client vertices are independent, with probabilities specified below.

For every integer $l\geq 0$, define a directed recursive block $B_l$.  Each block has a distinguished set of permanent vertices, denoted $P_l$.  These permanent vertices will be deterministic clients.  For $l=0$, the block $B_0$ consists of a single permanent vertex.

For $l\geq 1$, the block $B_l$ is obtained from two disjoint copies of $B_{l-1}$, denoted $B_{l-1}^{\mathrm L}$ and $B_{l-1}^{\mathrm R}$, together with a new midpoint vertex $m_l$.  Let
\[
    w_l := 2^{l-1}.
\]
For every permanent vertex $\ell\in P(B_{l-1}^{\mathrm L})$ and every permanent vertex $r'\in P(B_{l-1}^{\mathrm R})$, add a directed edge
$\ell \to r'$
of length $w_l$.  For every $r'\in P(B_{l-1}^{\mathrm R})$, add a directed edge
$r'\to m_l$
of length $w_l$.  For every $\ell\in P(B_{l-1}^{\mathrm L})$, add a directed edge
$m_l\to \ell$
of length $w_l$.  Thus, after contracting the two child blocks, the top level of $B_l$ is the directed triangle
\[
    B_{l-1}^{\mathrm L}
    \longrightarrow
    B_{l-1}^{\mathrm R}
    \longrightarrow
    m_l
    \longrightarrow
    B_{l-1}^{\mathrm L},
\]
where each displayed arc has length $w_l$.

The permanent set of $B_l$ is
$P_l := P(B_{l-1}^{\mathrm L})\cup P(B_{l-1}^{\mathrm R})$.
Thus $|P_l|=2^l$.  All permanent vertices are active with probability~$1$.  Every midpoint vertex is active independently with probability~$1/2$.

The full instance $I_l$ is obtained by adding a depot~$r$.  For every permanent vertex $\ell\in P_l$, add directed edges
$r\to \ell$ and $\ell\to r$,
each of length $w_l=2^{l-1}$.

Let $M_l$ be the set of midpoint vertices in $B_l$.  A midpoint of local depth $k$ has weight $w_k=2^{k-1}$.  There are $2^{l-k}$ midpoint vertices of local depth $k$ inside $B_l$.

\section{Statement}

\begin{theorem}\label{thm:recursive-gap}
For every $l\geq 1$, the instance $I_l$ satisfies
\[
    \optpost(I_l)=(3l+4)2^{l-2},
\]
and
\[
    \optadapt(I_l)=(l+1)2^l.
\]
Consequently,
\[
    \frac{\optadapt(I_l)}{\optpost(I_l)}
    =
    \frac{4(l+1)}{3l+4}.
\]
As $l\to\infty$, these ratios approach~$4/3$, so the asymmetric clairvoyance gap is at least~$4/3$.
\end{theorem}

The rest of this chapter proves the theorem.

\section{A posteriori cost}

We first analyze an open version of the block.  For a realization $A\subseteq M_l$ of the active midpoint vertices, let $P_l(A)$ be the minimum internal cost of a directed path that serves all deterministic permanent vertices of $B_l$ and all midpoint vertices in $A$, with free choice of the first and last permanent vertices of $B_l$.

For $l=0$, $P_0(A)=0$.

\begin{lemma}\label{lem:post-rec}
For $l\geq 1$,
\[
    P_l(A)
    =
    P_{l-1}(A_{\mathrm L})
    +
    P_{l-1}(A_{\mathrm R})
    +
    w_l\bigl(1+\1_{\{m_l\in A\}}\bigr),
\]
where $A_{\mathrm L}$ and $A_{\mathrm R}$ are the active midpoint sets inside the left and right child blocks.
\end{lemma}

\begin{proof}
All permanent vertices are deterministic, so any feasible path must serve both child blocks.  The internal cost in the two child blocks is at least
$P_{l-1}(A_{\mathrm L})+P_{l-1}(A_{\mathrm R})$.

If $m_l\notin A$, the cheapest top-level transition is $B_{l-1}^{\mathrm L}\to B_{l-1}^{\mathrm R}$, costing $w_l$.

If $m_l\in A$, the path must visit all three top-level objects.  The cheapest order is $B_{l-1}^{\mathrm R}\to m_l\to B_{l-1}^{\mathrm L}$, costing $2w_l$.
\end{proof}

Taking expectation with $\P[m_l\in A]=1/2$ gives
$\E[P_l(A)] = 2\E[P_{l-1}(A)]+\frac32 w_l$.
Iterating,
\[
    \E[P_l(A)]
    =
    \frac32\sum_{x\in M_l}w(x)
    = \frac32 \cdot l2^{l-1}
    = 3l2^{l-2}.
\]
In the full instance, the tour must enter the block from the depot and return
to the depot.  Since every permanent vertex has distance $w_l$ from and to the
depot, this adds exactly $2w_l=2^l$.  Therefore
\[
    \optpost(I_l)
    =
    2^l+3l2^{l-2}
    =
    (3l+4)2^{l-2}.
\]

\section{Adaptive cost}

We analyze the optimal adaptive policy via an open block cost $Q_l$.  Clearly $Q_0=0$.

\begin{lemma}[Three-state lemma]\label{lem:three-state}
Consider three top-level objects $L,R,M$.  The objects $L$ and $R$ are deterministic, while $M$ is active with probability~$1/2$.  The directed distances form the cycle $L\to R\to M\to L$, each arc of length~$w$.  The policy starts at either $L$ or $R$ (free), must serve both deterministic objects and call~$M$, and the process ends only once the policy is again at one of $L$ and~$R$.  Then the minimum expected cost is exactly $2w$.
\end{lemma}

\begin{proof}
A policy of expected cost $2w$ exists: start at $L$, serve $R$ (cost~$w$), call $M$; if active, travel $R\to M\to L$ (cost $2w$ with probability~$1/2$).  Expected: $w + \frac{1}{2}\cdot 2w = 2w$.

For the lower bound, if starting at $L$ and serving $R$ first: cost $\ge 1 + \frac{1}{2}(1+1) = 2$. If calling $M$ first: cost $\ge \frac{1}{2}\cdot 1 + \frac{1}{2}(2+2) > 2$. Starting at $R$ gives similar bounds. Rescaling gives $2w$.
\end{proof}

\begin{lemma}\label{lem:adapt-rec}
For every $l\geq 1$, $Q_l=2Q_{l-1}+2w_l$.
\end{lemma}

\begin{proof}
The top-level geometry is the three-state instance of Lemma~\ref{lem:three-state} with $w=w_l$.
Upper bound: use optimal three-state policy at top level, recurse in children; cost $2Q_{l-1}+2w_l$.
Lower bound: child costs $\ge 2Q_{l-1}$ (independence), top-level cost $\ge 2w_l$.
\end{proof}

Solving: $Q_l = l2^l$.
Adding the same depot contribution gives
$\optadapt(I_l) = 2^l + l2^l = (l+1)2^l$.

\section{The ratio}

\[
    \frac{\optadapt(I_l)}{\optpost(I_l)}
    =
    \frac{(l+1)2^l}{(3l+4)2^{l-2}}
    =
    \frac{4(l+1)}{3l+4}.
\]
This ratio approaches~$4/3$ as $l\to\infty$, so the supremum over this family is~$4/3$.
For $l=3$: $\optpost=26$, $\optadapt=32$, ratio $= 16/13 \approx 1.231$.
For $l=4$: $\optpost=64$, $\optadapt=80$, ratio $= 5/4$.

This completes the proof of Theorem~\ref{thm:recursive-gap}. \qed

\chapter{Approximate A Priori TSP via Dynamic Programming}\label{ch:dp}

We give a deterministic $(1+\varepsilon)$-approximation algorithm for
a~priori TSP on directed metrics.  For every fixed $\varepsilon>0$, its
running time is $c_\varepsilon^n\poly(n)$.

The algorithm is based on the following view of an a~priori order.  Once an
order $\pi=(r,v_1,\dots,v_n,r)$ is fixed, an edge from $v_i$ to $v_j$ is paid
exactly when $v_i$ and $v_j$ are active and all clients between them in the
order are inactive.  Thus the expected cost is a sum over ordered pairs.  The
coefficient of a pair contains the product
\[
    Q(i,j)=\prod_{i<k<j}(1-p_k),
\]
the probability that the interval between the two endpoints is completely
inactive.

Pairs with very small $Q(i,j)$ correspond to rare long shortcuts.  The first
part of the chapter proves that, up to a factor $1+O(\varepsilon)$, such pairs
may be discarded.  The proof does not try to charge each discarded pair
separately.  Instead, it slightly increases all activation probabilities.  This
adds enough active clients to split long jumps, while the directed triangle
inequality ensures that the shortcut cost can only increase.  This gives a
uniform bound comparing the full cost with the truncated pairwise cost.

It remains to minimize this truncated objective.  Directly remembering the
entire prefix of a partial order would be too expensive, so we group clients
into blocks: high-probability clients are kept as singletons, while
low-probability clients are grouped into small-mass blocks.  For each block we
precompute the best internal order.  Between two blocks, the truncated objective
depends only on the complete blocks between them.  Consequently, when the
dynamic program builds the block order from left to right, blocks that are
separated from the current end by inactivity product at most~$\eta$ can never
interact with future blocks and can be forgotten.

The rest of the chapter follows this outline: first the pairwise objective,
then the truncation lemma, then the block objective and its comparison with the
original order.  After that we give the dynamic program and choose the
parameters that yield the $(1+\varepsilon)$ guarantee.

\section{The pairwise objective}

Let
\[
    \pi=(r,v_1,\dots,v_n,r)
\]
be an a~priori order.  Put $v_0=v_{n+1}=r$ and $p_0=p_{n+1}=1$.  For brevity,
write
\[
    p_i:=p_{v_i},
    \qquad
    d_{ij}:=d(v_i,v_j).
\]
For $0\le i<j\le n+1$, define
\[
    Q(i,j):=\prod_{i<k<j}(1-p_k).
\]
Thus $Q(i,j)$ is the probability that every client strictly between $v_i$ and
$v_j$ is inactive.  The expected cost of $\pi$ is
\begin{equation}\label{eq:apriori-pairwise}
    \cost(\pi)
    =
    \sum_{0\le i<j\le n+1}p_i p_j Q(i,j)d_{ij}.
\end{equation}
Indeed, the term indexed by $(i,j)$ is paid exactly when $v_i$ and $v_j$ are
consecutive active vertices in the shortcut of~$\pi$.

Fix $0<\eta<1$.  We retain a pair $(i,j)$ when
$Q(i,j)>\eta$, and define
\begin{equation}\label{eq:local-pair-cost}
    \cost^{\loc}_{\eta}(\pi)
    :=
    \sum_{\substack{0\le i<j\le n+1\\Q(i,j)>\eta}}
    p_i p_j Q(i,j)d_{ij}.
\end{equation}

\section{\texorpdfstring{Truncating pairs with small \(Q\)}{Truncating pairs with small Q}}

We avoid choosing a splitting vertex separately for every discarded pair.
Instead, we slightly increase all activation probabilities.  The additional
active clients split all long jumps simultaneously, and the directed triangle
inequality shows that this can only increase the shortcut cost.

The parameter $\alpha$ measures the strength of this additional activation
round.  Write $q_v:=1-p_v$.  Besides the original active set, independently
activate $v$ with probability $1-q_v^\alpha$.  Then $v$ is absent from both
rounds with probability
\[
    q_v q_v^\alpha=q_v^{1+\alpha}.
\]
Consequently, a gap whose original inactivity probability is $Q(i,j)$ remains
empty after the additional round with probability
\[
    Q(i,j)^{1+\alpha}=Q(i,j)\,Q(i,j)^\alpha.
\]
Thus every discarded pair with $Q(i,j)\le\eta$ gains the uniform damping factor
$Q(i,j)^\alpha\le\eta^\alpha$.  The price is that each endpoint activation
probability grows by at most a factor $1+\alpha$.  The proof below formalizes
this tradeoff between the endpoint factor $(1+\alpha)^2$ and the damping factor
$\eta^\alpha$.

For an activation vector $s=(s_v)_{v\in V}$, let $\cost_s(\pi)$ denote the
expected cost of~$\pi$ under independent activation probabilities~$s_v$.
Thus $\cost_p(\pi)=\cost(\pi)$ for the original vector~$p$.

\begin{lemma}[Truncation by increasing activation probabilities]
\label{lem:long-gap-truncation}
Let $\alpha>0$, put $\kappa:=(1+\alpha)^2$, and suppose that
$0<\eta<1$ and $\kappa\eta^\alpha<1$.  Then every a~priori order~$\pi$
satisfies
\begin{equation}\label{eq:long-gap-truncation}
    \cost(\pi)
    \le
    \frac{\kappa}{1-\kappa\eta^\alpha}\,
    \cost^{\loc}_{\eta}(\pi).
\end{equation}
\end{lemma}

\begin{proof}
Define
\[
    \widetilde p_v:=1-(1-p_v)^{1+\alpha}.
\]
To couple the two activation vectors, first sample the original active set~$A$.
Independently, form a set~$B$ by including each client~$v$ with probability
$1-(1-p_v)^\alpha$.  Then $\widetilde A:=A\cup B$ has independent activation
probabilities~$\widetilde p_v$.

For every outcome, $A\subseteq\widetilde A$, and $\pi\vert_A$ is obtained from
$\pi\vert_{\widetilde A}$ by shortcutting vertices.  Hence the directed
triangle inequality gives
\[
    \cost(\pi\vert_A)\le \cost(\pi\vert_{\widetilde A}).
\]
Taking expectations,
\begin{equation}\label{eq:activation-monotonicity}
    \cost(\pi)\le \cost_{\widetilde p}(\pi).
\end{equation}

Convexity of $x\mapsto x^{1+\alpha}$ gives
\[
    \widetilde p_v
    =1-(1-p_v)^{1+\alpha}
    \le(1+\alpha)p_v.
\]
Moreover,
\[
    \widetilde Q(i,j)
    =\prod_{i<k<j}(1-\widetilde p_k)
    =Q(i,j)^{1+\alpha}.
\]
Thus the coefficient of $d_{ij}$ under~$\widetilde p$ is at most
\begin{align*}
    \widetilde p_i\widetilde p_j\widetilde Q(i,j)
    &\le
    \kappa p_ip_jQ(i,j)^{1+\alpha}\\
    &=\kappa Q(i,j)^\alpha\bigl(p_ip_jQ(i,j)\bigr).
\end{align*}

Set
\[
    T:=\cost(\pi),
    \qquad
    C:=\cost^{\loc}_{\eta}(\pi).
\]
On retained pairs, $Q(i,j)^\alpha\le1$; on discarded pairs,
$Q(i,j)^\alpha\le\eta^\alpha$.  Therefore
\[
    T
    \le \cost_{\widetilde p}(\pi)
    \le \kappa\bigl(C+\eta^\alpha(T-C)\bigr)
    \le \kappa C+\kappa\eta^\alpha T.
\]
Rearranging proves~\eqref{eq:long-gap-truncation}.
\end{proof}

\section{The block objective}

The dynamic program will order blocks rather than individual clients.  We
therefore define an objective on admissible ordered partitions.  For endpoints
in different blocks, this objective retains the exact inactivity factors of all
complete intervening blocks but replaces the factors contributed by vertices
inside the two endpoint blocks by~$1$.  This gives an upper bound on the
retained pair contribution and allows the contribution of a new block to be
evaluated from a bounded suffix of the preceding blocks.

Fix $0<\delta<1/4$.  A client is \emph{heavy} if
$p_v>\delta/2$ and \emph{light} otherwise.  For a set of clients~$B$, write
\[
    \mu(B):=\sum_{v\in B}p_v,
    \qquad
    q(B):=\prod_{v\in B}(1-p_v).
\]
For the depot block $\{r\}$, set $q(\{r\})=1$.

\begin{definition}[Admissible blocks and partitions]\label{def:admissible-partition}
A block is \emph{admissible} if it is either
\begin{enumerate}[label=(\alph*)]
    \item a singleton containing one heavy client, or
    \item a nonempty set of light clients with total mass at most~$\delta$.
\end{enumerate}
An ordered partition $(B_1,\dots,B_m)$ is admissible if all its blocks are
admissible and, whenever $B_{t-1}$ and $B_t$ are both light blocks,
\[
    \mu(B_t)\ge\delta/2.
\]
Thus, in every consecutive run of light blocks, only the first block may have
mass below~$\delta/2$.
\end{definition}

For a light block~$B$, define
\begin{equation}\label{eq:internal-block-cost}
    I(B)
    :=
    \min_{\sigma}
    \sum_{x<_\sigma y}p_xp_y d(x,y),
\end{equation}
where the minimum is over all orders~$\sigma$ of~$B$.  For a heavy singleton
and for the depot block, set $I=0$.  The values $I(B)$, together with a
minimizing order, can be computed by the recurrence
\begin{equation}\label{eq:internal-block-recurrence}
    I(B)
    =
    \min_{y\in B}
    \left(
        I(B\setminus\{y\})
        +\sum_{x\in B\setminus\{y\}}p_xp_y d(x,y)
    \right),
    \qquad I(\emptyset)=0.
\end{equation}
Indeed, the second term is exactly the contribution of the pairs whose last
vertex is~$y$.

Given an admissible ordered partition
$\mathcal B=(B_1,\dots,B_m)$, order each light block according to a minimizer in
\eqref{eq:internal-block-cost}, and concatenate the block orders.  Denote the
resulting a~priori order by~$\pi(\mathcal B)$.  Put
$B_0=B_{m+1}=\{r\}$ and $p_r=1$, and define
\[
    C(B_i,B_j)
    :=
    \sum_{x\in B_i}\sum_{y\in B_j}p_xp_y d(x,y).
\]
Define the \emph{block objective} by
\begin{equation}\label{eq:block-objective}
    F_\eta(\mathcal B)
    :=
    \sum_{t=1}^m I(B_t)
    +
    \sum_{\substack{0\le i<j\le m+1\\
                     \prod_{i<t<j}q(B_t)>\eta}}
    \left(\prod_{i<t<j}q(B_t)\right)C(B_i,B_j).
\end{equation}
The first sum accounts for pairs whose endpoints lie in the same block.  The
second sum accounts for pairs in different blocks whenever the inactivity
product of the complete intervening blocks exceeds~$\eta$.

\begin{lemma}[Comparison with the exact objective]\label{lem:block-comparison}
The block objective satisfies the following two inequalities.
\begin{enumerate}[label=(\roman*)]
    \item For every admissible ordered partition~$\mathcal B$,
    \begin{equation}\label{eq:block-objective-dominates-local}
        \cost^{\loc}_{\eta}(\pi(\mathcal B))
        \le F_\eta(\mathcal B).
    \end{equation}

    \item For every a~priori order~$\pi$, there is an admissible ordered
    partition $\mathcal B_\pi$ into consecutive sets of~$\pi$ such that
    \begin{equation}\label{eq:block-objective-from-order}
        F_\eta(\mathcal B_\pi)
        \le(1-\delta)^{-2}\cost(\pi).
    \end{equation}
\end{enumerate}
\end{lemma}

\begin{proof}
For~(i), first consider endpoints $x\in B_i$ and $y\in B_j$ with $i<j$.
In the order $\pi(\mathcal B)$, their exact coefficient is
\[
    p_xp_y\,
    q(B_i^{>x})\,
    \left(\prod_{i<t<j}q(B_t)\right)\,
    q(B_j^{<y}),
\]
where $B_i^{>x}$ denotes the vertices of $B_i$ after~$x$, and
$B_j^{<y}$ the vertices of $B_j$ before~$y$.  If $(x,y)$ is retained in
$\cost^{\loc}_{\eta}$, then its exact inactivity probability exceeds
$\eta$.  Therefore the product over the complete intervening blocks also
exceeds $\eta$, so the corresponding block pair occurs in
\eqref{eq:block-objective}.  Dropping the two endpoint factors can only
increase its coefficient.  If both endpoints lie in the same light block~$B$,
then their exact coefficient is at most $p_xp_y$, and the sum of these upper
bounds in the chosen internal order is exactly~$I(B)$.  Summing over all
retained pairs proves~\eqref{eq:block-objective-dominates-local}.

For~(ii), keep every heavy client as a singleton.  In each maximal consecutive
run of light clients in~$\pi$, scan from right to left.  Close a block as soon
as its mass reaches~$\delta/2$, and continue with the remaining prefix of the
run.  Every closed block has mass in $[\delta/2,\delta]$, because every light
client has probability at most~$\delta/2$.  The only possible block of mass
below~$\delta/2$ is the leftmost block of the run.  The resulting partition
$\mathcal B_\pi$ is therefore admissible.

Consider a cross-block term of $F_\eta(\mathcal B_\pi)$ with endpoints
$x\in B_i$ and $y\in B_j$, where $i<j$.  Relative to the exact coefficient of
this pair in~$\pi$, the corresponding term of $F_\eta(\mathcal B_\pi)$ omits
only the factors corresponding to the vertices after~$x$ in~$B_i$ and
before~$y$ in~$B_j$.  For a heavy singleton or
a depot block, the corresponding factor is~$1$.  For a light block~$B$,
\[
    q(B')\ge1-\sum_{v\in B'}p_v\ge1-\delta
    \qquad\text{for every }B'\subseteq B.
\]
Thus every cross-block coefficient in $F_\eta(\mathcal B_\pi)$ is at most
$(1-\delta)^{-2}$ times the corresponding exact coefficient in~$\pi$.

Finally, let $B$ be a light block, and retain the order induced by~$\pi$ on~$B$.
If $T_B(\pi)$ denotes the exact contribution of pairs with both endpoints
in~$B$, then
\[
    I(B)
    \le
    \sum_{\substack{x,y\in B\\x<_\pi y}}p_xp_y d(x,y)
    \le
    \frac{1}{1-\delta}T_B(\pi).
\]
The second inequality holds because every inactivity factor between two
vertices of~$B$ is at least $q(B)\ge1-\delta$.  Summing the cross-block and
internal comparisons proves~\eqref{eq:block-objective-from-order}.
\end{proof}

\section{The dynamic program}

We now minimize $F_\eta$ exactly over all admissible ordered partitions.
Consider a partial partition
\[
    (B_0,B_1,\dots,B_t),
    \qquad B_0=\{r\}.
\]
If
\[
    \prod_{s=1}^t q(B_s)>\eta,
\]
its \emph{relevant suffix} is the entire prefix
$(B_0,B_1,\dots,B_t)$.  Otherwise, let $a$ be the largest index such that
\[
    \prod_{s=a}^t q(B_s)\le\eta.
\]
Equivalently,
\[
    \prod_{s=a}^t q(B_s)\le\eta
    \quad\text{and}\quad
    \prod_{s=a+1}^t q(B_s)>\eta.
\]
The relevant suffix is then $(B_a,B_{a+1},\dots,B_t)$.

A state consists of the set $U\subseteq V$ of already placed clients and the
relevant suffix~$\Omega$ of their partial partition.  Its value is the minimum,
over all represented prefixes, of the internal terms of the placed blocks plus
all cross-block terms whose right endpoint has already been placed.  The
initial state is $(\emptyset,(B_0))$ with value~$0$.

A feasible next block~$B$ is either a singleton containing an unplaced heavy
client or a nonempty set of unplaced light clients with
$\mu(B)\le\delta$.  If the last block of the current prefix and~$B$ are both
light, we additionally require $\mu(B)\ge\delta/2$.  Suppose
$\Omega=(B_a,\dots,B_t)$.  Appending~$B$ adds
\begin{equation}\label{eq:dp-increment}
    I(B)
    +
    \sum_{s=a}^t
    \left(\prod_{s<u\le t}q(B_u)\right)C(B_s,B).
\end{equation}
Every block in the relevant suffix occurs in this sum, because the complete
blocks between it and~$B$ have inactivity product greater than~$\eta$.
After the transition, the suffix is trimmed according to the same rule.  Once
every client has been placed, the terminal depot block~$\{r\}$ is appended
using the cross-block part of~\eqref{eq:dp-increment} and internal cost zero.

\begin{lemma}[Correctness and running time of the DP]\label{lem:local-dp}
For fixed $\delta$ and~$\eta$, the dynamic program minimizes
$F_\eta$ over all admissible ordered partitions and runs in time
$c_{\delta,\eta}^n\poly(n)$.
\end{lemma}

\begin{proof}
When a block~$B$ is appended, the only new terms of~$F_\eta$ are its internal
term $I(B)$ and the cross-block terms whose right endpoint is~$B$.  These are
exactly the terms in~\eqref{eq:dp-increment}.

If a previous block is not in the relevant suffix, then the complete blocks
between it and the current end have inactivity product at most~$\eta$.
Appending more blocks only multiplies this product by numbers in~$[0,1]$.
Hence a forgotten block can never occur in a future term of~$F_\eta$.  The
state therefore contains precisely the information needed to evaluate every
continuation, and the recurrence is an exact shortest-path computation over all
admissible partial partitions.

It remains to bound the number of states.  We use the elementary inequality
\[
    \prod_{v\in S}(1-p_v)
    \le \exp\left(-\sum_{v\in S}p_v\right).
\]
If the whole prefix is retained, then its inactivity product exceeds
$\eta$, so its total activation mass is less than
$\log(1/\eta)$.  Otherwise, deleting the first retained block leaves an
inactivity product greater than~$\eta$, so the remaining blocks have total
activation mass less than $\log(1/\eta)$; the first retained block has mass at
most~$1$.  Thus the total activation mass of a relevant suffix is less than
\[
    M_\eta:=\log(1/\eta)+1.
\]

Every retained heavy singleton has mass greater than~$\delta/2$, and every
retained light block except the first one in a light run has mass at least
$\delta/2$.  Let $\mathcal G$ be the collection of these blocks.  Since their
total mass is less than $M_\eta$,
\[
    |\mathcal G|<2\frac{M_\eta}{\delta}.
\]
Every remaining retained block is the first light block of a light run.  The
number of light runs is at most one more than the number of retained heavy
blocks, and hence at most $|\mathcal G|+1$.  Therefore the number of retained
client blocks is at most
\[
    2|\mathcal G|+1
    <4\frac{M_\eta}{\delta}+1.
\]
Including the depot block, the relevant suffix has at most
\[
    K:=\left\lceil4\frac{M_\eta}{\delta}\right\rceil+2
    =O\!\left(\frac{\log(1/\eta)+1}{\delta}\right)
\]
blocks.  A state can be encoded by assigning each client one of the labels
``unplaced'', ``placed and forgotten'', or ``position $1,\dots,K$ in the
relevant suffix''.  Thus there are at most $(K+2)^n\poly(n)$ states.  A
transition may enumerate a subset of the unplaced light clients, contributing
at most another factor~$2^n$, and all numerical terms can be evaluated in
polynomial time.  This gives a running time
$c_{\delta,\eta}^n\poly(n)$.  The recurrence
\eqref{eq:internal-block-recurrence} for the values $I(B)$ takes
$O(n2^n)$ time and is absorbed in the same bound.
\end{proof}

\section{Approximation guarantee}

\begin{theorem}\label{thm:directed-apriori-ptas}
For every fixed $0<\varepsilon\le1$, a~priori TSP on directed
metrics has a deterministic algorithm running in time
$c_\varepsilon^n\poly(n)$ and returning an a~priori order~$\hat\pi$ such that
\[
    \cost(\hat\pi)\le(1+\varepsilon)\optprior.
\]
\end{theorem}

\begin{proof}
Set
\[
    \delta:=\frac{\varepsilon}{16},
    \qquad
    \alpha:=\frac{\varepsilon}{32},
    \qquad
    \eta:=\left(\frac{\varepsilon}{64}\right)^{32/\varepsilon}.
\]
Then
\[
    \eta^\alpha=\frac{\varepsilon}{64}.
\]

Let $\pi^*$ be an optimal a~priori order.  By
Lemma~\ref{lem:block-comparison}(ii), there is an admissible ordered partition
$\mathcal B^*$ into consecutive sets of~$\pi^*$ such that
\[
    F_\eta(\mathcal B^*)
    \le(1-\delta)^{-2}\cost(\pi^*)
    =(1-\delta)^{-2}\optprior.
\]
Let $\widehat{\mathcal B}$ be a partition minimizing $F_\eta$, as computed by
Lemma~\ref{lem:local-dp}, and put
$\hat\pi:=\pi(\widehat{\mathcal B})$.  Lemma~\ref{lem:block-comparison}(i)
gives
\[
    \cost^{\loc}_{\eta}(\hat\pi)
    \le F_\eta(\widehat{\mathcal B})
    \le F_\eta(\mathcal B^*)
    \le(1-\delta)^{-2}\optprior.
\]
Applying Lemma~\ref{lem:long-gap-truncation},
\begin{equation}\label{eq:final-approximation-chain}
    \cost(\hat\pi)
    \le
    \frac{(1+\alpha)^2}
         {1-(1+\alpha)^2\eta^\alpha}
    (1-\delta)^{-2}\optprior.
\end{equation}

Put $\kappa:=(1+\alpha)^2$.  Since $0<\varepsilon\le1$,
\[
    \kappa
    =1+\frac{\varepsilon}{16}+\frac{\varepsilon^2}{1024}
    \le1+\frac{\varepsilon}{8}
    \le\frac98.
\]
Therefore
\[
    \kappa\eta^\alpha
    =\kappa\frac{\varepsilon}{64}
    \le\frac{9\varepsilon}{512}
    <\frac{\varepsilon}{32}.
\]
Using $(1-x)^{-1}\le1+2x$ for $0\le x\le1/2$, we get
\begin{align*}
    \frac{\kappa}{1-\kappa\eta^\alpha}
    &\le
    \left(1+\frac{\varepsilon}{8}\right)
    \left(1+\frac{\varepsilon}{16}\right)\\
    &\le 1+\frac{25\varepsilon}{128}
    <1+\frac{\varepsilon}{4}.
\end{align*}
Also,
\[
    (1-\delta)^{-2}
    \le(1-2\delta)^{-1}
    =\frac{1}{1-\varepsilon/8}
    \le1+\frac{\varepsilon}{4}.
\]
Substituting these bounds into~\eqref{eq:final-approximation-chain} yields
\[
    \cost(\hat\pi)
    \le
    \left(1+\frac{\varepsilon}{4}\right)^2\optprior
    \le(1+\varepsilon)\optprior.
\]
Finally,
\[
    \log(1/\eta)
    =\frac{32}{\varepsilon}\log\frac{64}{\varepsilon},
\]
so $\delta$ and~$\eta$ depend only on~$\varepsilon$.  The running time remains
$c_\varepsilon^n\poly(n)$.
\end{proof}

\section{Open problems}

Several natural questions remain open.

\begin{enumerate}
    \item \textbf{Symmetric clairvoyance gap.}
    Is the clairvoyance gap for symmetric stochastic TSP at most $11/10$ (Conjecture~\ref{conj:main})?
    The three-chain construction shows that this would be tight, but a full proof or a counterexample is still missing.

    \item \textbf{Symmetric adaptivity gap.}
    Our lower bound of $5/4$ is the best known.  Is the adaptivity gap for symmetric metrics strictly larger?  What is its exact supremum?

    \item \textbf{Asymmetric clairvoyance gap.}
    Is the clairvoyance gap for asymmetric metrics $\Theta(1)$?  Our recursive construction gives a lower bound tending to~$4/3$.  Can this be matched by an $O(1)$ upper bound, or does the gap grow with~$n$?

    \item \textbf{Efficient adaptive policies.}
    Can one design polynomial-time adaptive policies whose expected cost is within a constant factor of~$\optadapt$ for asymmetric metrics?
    A promising direction is the backbone circulation approach: sample a backbone, construct a low-cost fractional circulation, round it to an integral assignment, and convert the result into an adaptive policy.

    \item \textbf{Improved a~priori algorithms.}
    Our $(1{+}\varepsilon)$-approximation for directed a~priori TSP runs in time $c_\varepsilon^n \poly(n)$, which is exponential.  Can polynomial-time algorithms with constant approximation ratios be obtained for directed a~priori TSP?
\end{enumerate}
